Cyklistika

🚴 Horská kola 200 km 📄 CSV se středníkem ⏱ Čas HH:MM:SS.d

Zadání

Vytvořte program v jazyce C, který načte soubor cyklistika.txt s výsledky závodníků na horských kolech (trasa 200 km), seřadí je podle dosažených časů a vypíše výsledkovou listinu.

Rozdíl oproti Jizerské 50: Místo národnosti (NOR, SWE) je zde pole klub — řetězec až 80 znaků. Formát záznamu je jednodušší — každé pole je přímý sloupec, žádný složený formát jako "Příjmení, Jméno (Ročník)".

Ukázka dat (cyklistika.txt)

cislo;Prijmeni;Jmeno;Rocnik;Klub;Cas
5001;FOJTIK;Jan;2008;FORCE KCK Zlin;9:23:45.4
5002;JOBANEK;Jan;2003;Superior Suninvent Team;9:19:09.9
5003;STASTNY;Petr;2008;RAJEC TREK;9:25:34.4
...

Kompletní řešení

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define MAX_CYCLISTS 200
#define MAX_NAME     50
#define MAX_CLUB     80
#define BUFFER_SIZE  256

typedef struct {
    int  number;
    char surname[MAX_NAME];
    char name[MAX_NAME];
    int  year;
    char club[MAX_CLUB];
    int  time_sec;
    char time_str[20];
} Cyclist;

/* "9:19:09.9" → sekundy (desetiny ignorujeme, stačí pro řazení) */
int time_to_seconds(const char *s) {
    int h, m, sec, ds;
    sscanf(s, "%d:%d:%d.%d", &h, &m, &sec, &ds);
    return h * 3600 + m * 60 + sec;
}

void swap(Cyclist *a, Cyclist *b) {
    Cyclist tmp = *a; *a = *b; *b = tmp;
}

void sort_by_time(Cyclist *arr, int n) {
    for (int i = 0; i < n - 1; i++)
        for (int j = 0; j < n - 1 - i; j++)
            if (arr[j].time_sec > arr[j+1].time_sec)
                swap(&arr[j], &arr[j+1]);
}

int read_data(const char *filename, Cyclist *arr) {
    FILE *f = fopen(filename, "r");
    if (!f) { printf("Chyba: nelze otevrit '%s'\n", filename); return 0; }

    char buf[BUFFER_SIZE];
    int n = 0;

    fgets(buf, sizeof(buf), f);  /* přeskočit hlavičku */

    /* Formát: cislo;Prijmeni;Jmeno;Rocnik;Klub;Cas */
    while (fgets(buf, sizeof(buf), f) && n < MAX_CYCLISTS) {
        buf[strcspn(buf, "\n")] = 0;

        char *p = strtok(buf, ";");
        arr[n].number = atoi(p);

        p = strtok(NULL, ";");
        strcpy(arr[n].surname, p);

        p = strtok(NULL, ";");
        strcpy(arr[n].name, p);

        p = strtok(NULL, ";");
        arr[n].year = atoi(p);

        p = strtok(NULL, ";");
        strcpy(arr[n].club, p);

        p = strtok(NULL, ";");
        strcpy(arr[n].time_str, p);
        arr[n].time_sec = time_to_seconds(p);

        n++;
    }
    fclose(f);
    return n;
}

void print_results(Cyclist *arr, int n) {
    printf("\nVYSLEDKOVA LISTINA - CYKLISTIKA 200 KM\n");
    printf("=======================================================================\n");
    printf("Por. | Cislo | Prijmeni    | Jmeno | Klub                   | Cas\n");
    printf("-----+-------+-------------+-------+------------------------+-----------\n");
    for (int i = 0; i < n; i++)
        printf("%3d. | %5d | %-11s | %-5s | %-22s | %s\n",
               i+1, arr[i].number, arr[i].surname,
               arr[i].name, arr[i].club, arr[i].time_str);
    printf("=======================================================================\n\n");
}

int main(void) {
    Cyclist cyclists[MAX_CYCLISTS];
    int n = read_data("cyklistika.txt", cyclists);
    if (n == 0) return 1;
    sort_by_time(cyclists, n);
    print_results(cyclists, n);
    return 0;
}
Klíčové body:
• Parsování je nejjednodušší z těchto čtyř úloh — každé pole je přímo oddělené ;, žádné vnořené formáty
• MAX_CLUB 80 — klub může být dlouhý řetězec, např. "Cyklobulis Bystrice pod Hostynem"
• time_to_seconds() ignoruje desetiny — pro řazení stačí přesnost na sekundu
• Struktura neobsahuje country jako Jizerská 50, ale club — jinak je kód totožný

Co se naučíš

  • Přímé CSV parsování — nejčistší forma, každý strtok() = jeden sloupec
  • Dlouhé řetězce — club[MAX_CLUB] musí být dost velké pro všechny názvy klubů
  • Čas HH:MM:SS.d — čtyři čísla v sscanf, desetiny ignorujeme pro řazení
  • Stejný vzor jako Jizerská 50 — po pochopení jedné úlohy je druhá přímočará
