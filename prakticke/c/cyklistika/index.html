Cyklistika

🚴 Horská kola 200 km 📊 CSV parsing + řazení

Zadání

Vytvořte program v jazyce C, který s využitím vzorového datového souboru přihlášených závodníků na horských kolech na trase 200 km cyklistika.txt vypíše na obrazovku výsledkovou listinu závodu.

Úloha je podobná Jizerské 50, ale se středníkem jako separátorem a s klubem místo státu.

Kompletní řešení (C)

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define MAX_CYCLISTS 200
#define MAX_NAME 50
#define MAX_CLUB 80
#define BUFFER_SIZE 256

typedef struct {
  int number;           // Startovní číslo
  char surname[MAX_NAME];
  char name[MAX_NAME];
  int year;
  char club[MAX_CLUB];
  int time_sec;         // Čas v sekundách
  char time_str[20];    // Čas jako string
} Cyclist;

// Funkce: převést "HH:MM:SS.d" na sekundy
int time_to_seconds(const char *time_str) {
  int h, m, s, ds;
  sscanf(time_str, "%d:%d:%d.%d", &h, &m, &s, &ds);
  return h * 3600 + m * 60 + s;
}

// Funkce: výměna
void swap(Cyclist *a, Cyclist *b) {
  Cyclist temp = *a;
  *a = *b;
  *b = temp;
}

// Funkce: bubble sort
void sort_by_time(Cyclist *cyclists, int count) {
  for (int i = 0; i < count - 1; i++) {
    for (int j = 0; j < count - 1 - i; j++) {
      if (cyclists[j].time_sec > cyclists[j + 1].time_sec) {
        swap(&cyclists[j], &cyclists[j + 1]);
      }
    }
  }
}

// Funkce: čtení dat
int read_data(const char *filename, Cyclist *cyclists) {
  FILE *file = fopen(filename, "r");
  if (file == NULL) {
    printf("Chyba: Nepodařilo se otevřít soubor '%s'.\n", filename);
    return 0;
  }

  char buffer[BUFFER_SIZE];
  int count = 0;

  // Přeskočit hlavičku
  fgets(buffer, BUFFER_SIZE, file);

  // Čtení řádků
  // Formát: cislo;Prijmeni;Jmeno;Rocnik;Klub;Cas
  while (fgets(buffer, BUFFER_SIZE, file) && count < MAX_CYCLISTS) {
    buffer[strcspn(buffer, "\n")] = 0;

    char *part;

    // Číslo
    part = strtok(buffer, ";");
    cyclists[count].number = atoi(part);

    // Příjmení
    part = strtok(NULL, ";");
    strcpy(cyclists[count].surname, part);

    // Jméno
    part = strtok(NULL, ";");
    strcpy(cyclists[count].name, part);

    // Ročník
    part = strtok(NULL, ";");
    cyclists[count].year = atoi(part);

    // Klub
    part = strtok(NULL, ";");
    strcpy(cyclists[count].club, part);

    // Čas
    part = strtok(NULL, ";");
    strcpy(cyclists[count].time_str, part);
    cyclists[count].time_sec = time_to_seconds(part);

    count++;
  }

  fclose(file);
  return count;
}

// Funkce: výpis
void print_results(Cyclist *cyclists, int count) {
  printf("\n");
  printf("VÝSLEDKOVÁ LISTINA - CYKLISTIKA 200 KM\n");
  printf("===================================================================\n");
  printf("Por. | Číslo | Příjmení         | Jméno       | Klub              | Čas       \n");
  printf("-----+-------+------------------+-------------+-------------------+----------\n");

  for (int i = 0; i < count; i++) {
    printf("%3d. | %5d | %-16s | %-11s | %-17s | %s\n",
           i + 1,
           cyclists[i].number,
           cyclists[i].surname,
           cyclists[i].name,
           cyclists[i].club,
           cyclists[i].time_str);
  }
  printf("===================================================================\n\n");
}

int main(void) {
  Cyclist cyclists[MAX_CYCLISTS];
  int count;

  count = read_data("cyklistika.txt", cyclists);

  if (count == 0) {
    return 1;
  }

  // Seřadit
  sort_by_time(cyclists, count);

  // Vypsat
  print_results(cyclists, count);

  return 0;
}
Struktura: Místo státu (NOR, SWE) máme klub (string až 80 znaků). Zbytek je identické s Jizerskou 50.
