Rychlobruslení

⛸️ 10 000 metrů 📄 Mezerami oddělené + závorky ⏱ Čas MM:SS.CS

Zadání

Vytvořte program v jazyce C, který načte soubor rychlobrusleni10000m.txt s výsledky závodníků na 10 000 metrů, seřadí je jednotlivě podle dosažených časů a vypíše výsledkovou listinu.

Nejtěžší parsování ze čtyř úloh. Soubor není klasické CSV — hodnoty jsou odděleny mezerami a státní kód je v závorkách: 1 Riccardo Lorello (ITA) 12:56.22. Závodníci startují v párech, ale řadíme je jednotlivě.

Ukázka dat (rychlobrusleni10000m.txt)

cislo_dvojice jmeno prijmeni (stat) cas
1 Riccardo Lorello (ITA) 12:56.22
1 Hald_Viktor Thorup (DEN) 59:59:59
2 Sigurd Henriksen (NOR) 13:27.29
2 Bart Swings (BEL) 13:05.60
3 Stijn van_de_Bunt (NED) 12:45.75
5 Metodej Jilek (CZE) 12:33.43
...
Přístup k parsování:
1. Přečti číslo páru pomocí sscanf()
2. Přejdi dopředu v bufferu, dokud nenarazíš na začátek jména (první velké písmeno)
3. Zkopíruj zbytek řádku, najdi ( závorku → oddělíš jméno + příjmení od státu
4. Stát přečti z obsahu závorky, čas ze konce řádku za )

Kompletní řešení

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define MAX_RACERS  100
#define MAX_NAME    50
#define BUFFER_SIZE 256

typedef struct {
    int  pair;
    char name[MAX_NAME];
    char surname[MAX_NAME];
    char country[4];    /* 3 znaky + '\0' */
    int  time_sec;
    char time_str[20];
} Racer;

/* "12:33.43" → sekundy (setiny ignorujeme, stačí pro řazení) */
int time_to_seconds(const char *s) {
    int m, sec, cs;
    sscanf(s, "%d:%d.%d", &m, &sec, &cs);
    return m * 60 + sec;
}

void swap(Racer *a, Racer *b) {
    Racer tmp = *a; *a = *b; *b = tmp;
}

void sort_by_time(Racer *arr, int n) {
    for (int i = 0; i < n - 1; i++)
        for (int j = 0; j < n - 1 - i; j++)
            if (arr[j].time_sec > arr[j+1].time_sec)
                swap(&arr[j], &arr[j+1]);
}

int read_data(const char *filename, Racer *arr) {
    FILE *f = fopen(filename, "r");
    if (!f) { printf("Chyba: nelze otevrit '%s'\n", filename); return 0; }

    char buf[BUFFER_SIZE];
    int n = 0;

    fgets(buf, sizeof(buf), f);  /* přeskočit hlavičku */

    /* Formát: cislo_dvojice Jmeno Prijmeni (STAT) cas */
    while (fgets(buf, sizeof(buf), f) && n < MAX_RACERS) {
        buf[strcspn(buf, "\n")] = 0;

        /* číslo páru */
        if (sscanf(buf, "%d", &arr[n].pair) != 1) continue;

        /* přejdi na první velké písmeno (začátek jména) */
        char *p = buf;
        while (*p && !(*p >= 'A' && *p <= 'Z')) p++;
        if (!*p) continue;

        /* zkopíruj "Jmeno Prijmeni (STAT) cas" */
        char tmp[BUFFER_SIZE];
        strcpy(tmp, p);

        /* najdi otevírací závorku */
        char *paren = strchr(tmp, '(');
        if (!paren) continue;

        /* text před závorkou je "Jmeno Prijmeni " */
        *(paren - 1) = '\0';
        char *last_sp = strrchr(tmp, ' ');
        if (last_sp) {
            strcpy(arr[n].surname, last_sp + 1);  /* příjmení = poslední slovo */
            *last_sp = '\0';
            strcpy(arr[n].name, tmp);             /* jméno = zbytek */
        }

        /* stát: 3 znaky uvnitř závorky */
        sscanf(paren + 1, "%3s", arr[n].country);

        /* čas: text za ')' na původním buf */
        char *time_p = strrchr(buf, ')');
        if (time_p) {
            time_p++;
            while (*time_p == ' ') time_p++;
            strcpy(arr[n].time_str, time_p);
            arr[n].time_sec = time_to_seconds(time_p);
        }

        n++;
    }
    fclose(f);
    return n;
}

void print_results(Racer *arr, int n) {
    printf("\nRYCHLOBRUSLENI - 10 000 m\n");
    printf("=========================================================\n");
    printf("Por. | Jmeno            | Prijmeni        | Stat | Cas\n");
    printf("-----+------------------+-----------------+------+----------\n");
    for (int i = 0; i < n; i++)
        printf("%3d. | %-16s | %-15s | %s   | %s\n",
               i+1, arr[i].name, arr[i].surname,
               arr[i].country, arr[i].time_str);
    printf("=========================================================\n\n");
}

int main(void) {
    Racer racers[MAX_RACERS];
    int n = read_data("rychlobrusleni10000m.txt", racers);
    if (n == 0) return 1;
    sort_by_time(racers, n);
    print_results(racers, n);
    return 0;
}
Klíčové triky parsování:
• Místo strtok() prohledáváme buffer ručně pomocí pointeru p
• while (*p && !(*p >= 'A' && *p <= 'Z')) p++ — přejdi na první velké písmeno
• strchr(tmp, '(') najde závorku → odtud oddělíme jméno/příjmení od státu
• strrchr(tmp, ' ') najde poslední mezeru → oddělí příjmení od jména
• sscanf(paren + 1, "%3s", country) — musí být %3s (3 znaky pro "ITA") a country[4]

Co se naučíš

  • Ruční procházení bufferu — pointer aritmetika místo strtok()
  • strchr() a strrchr() — hledání prvního/posledního výskytu znaku
  • Závorky jako oddělovač — extrakce obsahu závorky pomocí (paren + 1)
  • country[4] — 3-znakový kód potřebuje 4 bajty s ukončovacím '\0'
  • %3s v sscanf — limit délky načteného řetězce
