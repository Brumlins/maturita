Rychlobruslení

⛸️ 10 000 metrů 📊 Parsování + řazení

Zadání

Vytvořte program v jazyce C, který s využitím datového souboru rychlobrusleni10000m.txt zpracuje výsledky rychlobruslařů na 10 000 metrů a vypíše výsledkovou listinu seřazenou podle dosažených časů.

Zvláštnost: Závodníci jsou v souborů v párech (skupinách), ale řadíme jednotlivě podle času.

Kompletní řešení (C)

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define MAX_RACERS 100
#define MAX_NAME 50
#define BUFFER_SIZE 256

typedef struct {
  int pair_number;      // Číslo skupiny/páru
  char name[MAX_NAME];  // Jméno
  char surname[MAX_NAME]; // Příjmení
  char country[3];      // Státní kód
  int time_sec;         // Čas v sekundách
  char time_str[20];    // Čas jako string (MM:SS.MS)
} Racer;

// Funkce: převést "MM:SS.MS" na sekundy
int time_to_seconds(const char *time_str) {
  int m, s, ms;
  sscanf(time_str, "%d:%d.%d", &m, &s, &ms);
  return m * 60 + s;
}

// Funkce: výměna
void swap(Racer *a, Racer *b) {
  Racer temp = *a;
  *a = *b;
  *b = temp;
}

// Funkce: bubble sort
void sort_by_time(Racer *racers, int count) {
  for (int i = 0; i < count - 1; i++) {
    for (int j = 0; j < count - 1 - i; j++) {
      if (racers[j].time_sec > racers[j + 1].time_sec) {
        swap(&racers[j], &racers[j + 1]);
      }
    }
  }
}

// Funkce: čtení dat
int read_data(const char *filename, Racer *racers) {
  FILE *file = fopen(filename, "r");
  if (file == NULL) {
    printf("Chyba: Nepodařilo se otevřít '%s'.\n", filename);
    return 0;
  }

  char buffer[BUFFER_SIZE];
  int count = 0;

  // Přeskočit hlavičku
  fgets(buffer, BUFFER_SIZE, file);

  // Čtení řádků
  // Formát: cislo_dvojice jmeno prijmeni (stat) cas
  while (fgets(buffer, BUFFER_SIZE, file) && count < MAX_RACERS) {
    buffer[strcspn(buffer, "\n")] = 0;

    // Číslo páru
    int pair;
    if (sscanf(buffer, "%d", &pair) != 1) continue;
    racers[count].pair_number = pair;

    // Zbytek: "Jmeno Prijmeni (STAT) CAS"
    char *start = buffer;
    while (*start && !(' ' && *(start+1) && *(start+1) >= 'A')) start++;
    while (*start == ' ') start++;

    // Parsování jména a příjmení
    char temp[256];
    strcpy(temp, start);

    // Najít závorku pro stát
    char *paren_start = strchr(temp, '(');
    if (!paren_start) continue;

    // Příjmení je poslední slovo před závorkou
    *(paren_start - 1) = 0;
    char *last_space = strrchr(temp, ' ');
    if (last_space) {
      strcpy(racers[count].surname, last_space + 1);
      *last_space = 0;
      strcpy(racers[count].name, temp);
    }

    // Stát
    paren_start++;
    sscanf(paren_start, "%2s", racers[count].country);

    // Čas (posledních 8 znaků)
    char *time_pos = strrchr(buffer, ')');
    if (time_pos) {
      time_pos++;
      while (*time_pos == ' ') time_pos++;
      strcpy(racers[count].time_str, time_pos);
      racers[count].time_sec = time_to_seconds(time_pos);
    }

    count++;
  }

  fclose(file);
  return count;
}

// Funkce: výpis
void print_results(Racer *racers, int count) {
  printf("\n");
  printf("RYCHLOBRUSLENÍ - 10 000 m\n");
  printf("=========================================================\n");
  printf("Por. | Jméno               | Příjmení         | Stát | Čas       \n");
  printf("-----+---------------------+------------------+------+----------\n");

  for (int i = 0; i < count; i++) {
    printf("%3d. | %-19s | %-16s | %s   | %s\n",
           i + 1,
           racers[i].name,
           racers[i].surname,
           racers[i].country,
           racers[i].time_str);
  }
  printf("=========================================================\n\n");
}

int main(void) {
  Racer racers[MAX_RACERS];
  int count;

  count = read_data("rychlobrusleni10000m.txt", racers);

  if (count == 0) {
    return 1;
  }

  // Seřadit
  sort_by_time(racers, count);

  // Vypsat
  print_results(racers, count);

  return 0;
}
Poznámka: Parsování je složitější kvůli formátu se státem v závorce. Přístup: (1) Najít pozici závorky, (2) Parsovat stát, (3) Zbývající text rozdělit na jméno a příjmení.
