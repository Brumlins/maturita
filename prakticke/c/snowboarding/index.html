Snowboarding

🏂 Paralelní obří slalom 📄 CSV s pipe separátorem ⏱ Dvě kola + součet

Zadání

Vytvořte program v jazyce C, který načte soubor snowboarding.txt s výsledky závodnic paralelního obřího slalomu, vypočítá celkový čas (součet obou kol) a vypíše výsledkovou listinu seřazenou vzestupně.

Klíčový rozdíl oproti Jizerské 50: Dvě kola, časy se sčítají. Časy kol jsou ve formátu SS.CS (sekundy.setiny), celkový čas MM:SS.CS. Separátor je | (pipe), ne středník.

Ukázka dat (snowboarding.txt)

Jmeno|Zeme|Modra draha|Cervena draha|Celkovy cas
Ester Ledecka|CZE|45.04|45.75|1:30.79
Zuzana Maderova|CZE|45.38|46.10|1:31.48
Julie Zogg|SUI|47.05|47.33|1:34.38
...
Formát časů: Kola jsou v sekundách — 45.04 = 45 sekund a 4 setiny. Celkový čas je v minutách — 1:30.79 = 1 minuta, 30 sekund, 79 setin. Pro přesné řazení převádíme vše na setiny sekundy (centisekundy).

Kompletní řešení

#include <stdatomic.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define VSTUP "snowboarding.txt"
#define ODDELOVACE "|.:"
#define SIZE_STRING 30
#define SIZE_RADEK 100


typedef struct
{
    int minuty;
    int sekundy;
    int setiny;
}CAS;

typedef struct
{
    CAS modra;
    CAS cervena;
}JIZDY;

typedef struct
{
    char jmeno[SIZE_STRING];
    char zeme[SIZE_STRING];
    JIZDY drahy;
    CAS celkovy_cas;
}ZAVODNIK;

ZAVODNIK *parseFile(int *pocet)
{
    FILE *fr;
    char radek[SIZE_RADEK];
    char *pch;
    int i = 0;
    int castRadku;
    ZAVODNIK *zavodnik = NULL;
    ZAVODNIK *viceZavodniku = NULL;

    fr = fopen(VSTUP,"r");
    if (fr == NULL)
    {
        printf("Soubor %s se nepodarilo uspesne otevrit! ", VSTUP);
        return NULL;
    }
    while (fgets(radek,SIZE_RADEK,fr)!= NULL)
    {
        if (i != 0)
        {
            viceZavodniku = (ZAVODNIK *) realloc(zavodnik, i * sizeof(ZAVODNIK));

            if (viceZavodniku != NULL)
            {
                zavodnik = viceZavodniku;
                castRadku = 0;
                pch = strtok(radek, ODDELOVACE);
                while (pch != NULL)
                {
                    switch (castRadku)
                    {
                    case 0:
                        strcpy(zavodnik[i - 1].jmeno,pch);
                        break;
                    case 1:
                        strcpy(zavodnik[i - 1].zeme,pch);
                        break;
                    case 2:
                        zavodnik[i - 1].drahy.modra.sekundy = atoi(pch);
                        break;
                    case 3:
                        zavodnik[i - 1].drahy.modra.setiny = atoi(pch);
                        break;
                    case 4:
                        zavodnik[i - 1].drahy.cervena.sekundy = atoi(pch);
                        break;
                    case 5:
                        zavodnik[i - 1].drahy.cervena.setiny = atoi(pch);
                        break;
                    case 6:
                        zavodnik[i-1].celkovy_cas.minuty = atoi(pch);
                        break;
                    case 7:
                        zavodnik[i-1].celkovy_cas.sekundy = atoi(pch);
                        break;
                    case 8:
                        zavodnik[i - 1].celkovy_cas.setiny = atoi(pch);
                        break;
                    }
                    pch = strtok(NULL,ODDELOVACE);
                    castRadku++;
                }
            }else
            {
                i = 0;
                free(zavodnik);
                printf("Chyba pri realokaci pameti! ");
                break;
            }
        }
        i++;
    }
    if (fclose(fr) == EOF)
    {
        printf("Soubor %s se nepodarilo uspesne zavrit! ",VSTUP);
    }
    *pocet = i - 1;
    return zavodnik;
}

int cas_na_setiny(CAS cas)
{
    return cas.minuty * 6000 + cas.sekundy * 100 + cas.setiny;
}


void swap(ZAVODNIK *a, ZAVODNIK *b)
{
    ZAVODNIK tmp;
    tmp = *a;
    *a = *b;
    *b = tmp;
}

void bubbleSort(ZAVODNIK *zavodnicek, int pocet)
{
    for (int i = 0; i < pocet; i++)
    {
        for (int j = 0; j < pocet - i - 1; j++)
        {
            if (cas_na_setiny(zavodnicek[j].celkovy_cas) > cas_na_setiny(zavodnicek[j+1].celkovy_cas))
            {
                swap(&zavodnicek[j],&zavodnicek[j+1]);
            }
        }
    }
}

void vypis(ZAVODNIK *zavodnik, int pocet)
{
    printf("SNOWBOARD ZOH 2026  -  KVALIFIKACE paralerni obri slalom zeny\n");
    printf("----------------------------------------------------------------------------\n");
    printf("poradi          jmeno    zeme   modra draha    cervena draha   celkem\n");
    printf("----------------------------------------------------------------------------\n");
    for (int i = 0; i < pocet; i++)
    {
        printf("%5d  %20s  %5s  %.2d.%.2d     %.2d.%.2d   %d:%.2d.%.2d\n",
        i + 1, zavodnik[i].jmeno,zavodnik[i].zeme,zavodnik[i].drahy.modra.sekundy,zavodnik[i].drahy.modra.setiny,
        zavodnik[i].drahy.cervena.sekundy,zavodnik[i].drahy.cervena.setiny,zavodnik[i].celkovy_cas.minuty,
        zavodnik[i].celkovy_cas.sekundy,zavodnik[i].celkovy_cas.setiny);
    }
    printf("----------------------------------------------------------------------------\n");
}

int main(void)
{
    ZAVODNIK *zavodnik;
    int pocet;
    zavodnik = parseFile(&pocet);
    bubbleSort(zavodnik,pocet);
    vypis(zavodnik,pocet);
    free(zavodnik);
    return 0;
}
Proč funkce to_cs() kontroluje dvojtečku:
Kola mají formát SS.CS (např. 45.04) — žádná minuta.
Celkový čas má formát MM:SS.CS (např. 1:30.79) — je přes minutu.
Funkce zjistí formát pomocí strchr(s, ':') a použije správný sscanf.

Celkový čas nepřebíráme ze souboru — počítáme ho sami: total = kolo1 + kolo2, pak převedeme zpět na string pro výpis.

Co se naučíš

  • Pipe separátor — strtok(buf, "|") místo středníku
  • Dva formáty času — detekce strchr(s, ':') a správný sscanf
  • Práce s centisekundami — přesné sčítání a konverze zpět na string
  • Výpočet v strukturách — total_cs se nevytahuje ze souboru, počítá se
  • sprintf() pro formátování čísla do stringu
