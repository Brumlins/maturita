Snowboarding

🏂 Paralelní obří slalom 📄 CSV s pipe separátorem ⏱ Dvě kola + součet

Zadání

Vytvořte program v jazyce C, který načte soubor snowboarding.txt s výsledky závodnic paralelního obřího slalomu, vypočítá celkový čas (součet obou kol) a vypíše výsledkovou listinu seřazenou vzestupně.

Klíčový rozdíl oproti Jizerské 50: Dvě kola, časy se sčítají. Časy kol jsou ve formátu SS.CS (sekundy.setiny), celkový čas MM:SS.CS. Separátor je | (pipe), ne středník.

Ukázka dat (snowboarding.txt)

Jmeno|Zeme|Modra draha|Cervena draha|Celkovy cas
Ester Ledecka|CZE|45.04|45.75|1:30.79
Zuzana Maderova|CZE|45.38|46.10|1:31.48
Julie Zogg|SUI|47.05|47.33|1:34.38
...
Formát časů: Kola jsou v sekundách — 45.04 = 45 sekund a 4 setiny. Celkový čas je v minutách — 1:30.79 = 1 minuta, 30 sekund, 79 setin. Pro přesné řazení převádíme vše na setiny sekundy (centisekundy).

Kompletní řešení

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define MAX_RACERS  50
#define MAX_NAME    100
#define BUFFER_SIZE 256

typedef struct {
    char name[MAX_NAME];
    char country[4];    /* 3 znaky + '\0' */
    int  time1_cs;      /* kolo 1 v centisekundách */
    int  time2_cs;      /* kolo 2 v centisekundách */
    int  total_cs;      /* součet pro řazení */
    char time1_str[20];
    char time2_str[20];
    char total_str[20];
} Racer;

/*
 * Převod na centisekundy — zvládá oba formáty:
 *   "45.04"   → SS.CS  (kola)
 *   "1:30.79" → MM:SS.CS (celkový čas)
 */
int to_cs(const char *s) {
    int m = 0, sec = 0, cs = 0;
    if (strchr(s, ':'))
        sscanf(s, "%d:%d.%d", &m, &sec, &cs);
    else
        sscanf(s, "%d.%d", &sec, &cs);
    return m * 6000 + sec * 100 + cs;
}

/* Zpět na string "MM:SS.CS" */
void cs_to_str(int total, char *out) {
    int m   = total / 6000;
    int sec = (total % 6000) / 100;
    int cs  = total % 100;
    sprintf(out, "%d:%02d.%02d", m, sec, cs);
}

void swap(Racer *a, Racer *b) {
    Racer tmp = *a; *a = *b; *b = tmp;
}

void sort_by_time(Racer *arr, int n) {
    for (int i = 0; i < n - 1; i++)
        for (int j = 0; j < n - 1 - i; j++)
            if (arr[j].total_cs > arr[j+1].total_cs)
                swap(&arr[j], &arr[j+1]);
}

int read_data(const char *filename, Racer *arr) {
    FILE *f = fopen(filename, "r");
    if (!f) { printf("Chyba: nelze otevrit '%s'\n", filename); return 0; }

    char buf[BUFFER_SIZE];
    int n = 0;

    fgets(buf, sizeof(buf), f);  /* přeskočit hlavičku */

    /* Formát: Jmeno|Zeme|Modra|Cervena|Celkovy */
    while (fgets(buf, sizeof(buf), f) && n < MAX_RACERS) {
        buf[strcspn(buf, "\n")] = 0;

        char *p = strtok(buf, "|");
        strcpy(arr[n].name, p);

        p = strtok(NULL, "|");
        while (*p == ' ') p++;
        strcpy(arr[n].country, p);

        /* kolo 1 */
        p = strtok(NULL, "|");
        while (*p == ' ') p++;
        strcpy(arr[n].time1_str, p);
        arr[n].time1_cs = to_cs(p);

        /* kolo 2 */
        p = strtok(NULL, "|");
        while (*p == ' ') p++;
        strcpy(arr[n].time2_str, p);
        arr[n].time2_cs = to_cs(p);

        /* celkový čas = součet obou kol */
        arr[n].total_cs = arr[n].time1_cs + arr[n].time2_cs;
        cs_to_str(arr[n].total_cs, arr[n].total_str);

        n++;
    }
    fclose(f);
    return n;
}

void print_results(Racer *arr, int n) {
    printf("\nSNOWBOARD - KVALIFIKACE zeny\n");
    printf("================================================================\n");
    printf("Por. | Jmeno           | Zeme | Kolo 1 | Kolo 2 | Celkem\n");
    printf("-----+-----------------+------+--------+--------+----------\n");
    for (int i = 0; i < n; i++)
        printf("%3d. | %-15s | %s   | %6s | %6s | %s\n",
               i+1, arr[i].name, arr[i].country,
               arr[i].time1_str, arr[i].time2_str, arr[i].total_str);
    printf("================================================================\n\n");
}

int main(void) {
    Racer racers[MAX_RACERS];
    int n = read_data("snowboarding.txt", racers);
    if (n == 0) return 1;
    sort_by_time(racers, n);
    print_results(racers, n);
    return 0;
}
Proč funkce to_cs() kontroluje dvojtečku:
Kola mají formát SS.CS (např. 45.04) — žádná minuta.
Celkový čas má formát MM:SS.CS (např. 1:30.79) — je přes minutu.
Funkce zjistí formát pomocí strchr(s, ':') a použije správný sscanf.

Celkový čas nepřebíráme ze souboru — počítáme ho sami: total = kolo1 + kolo2, pak převedeme zpět na string pro výpis.

Co se naučíš

  • Pipe separátor — strtok(buf, "|") místo středníku
  • Dva formáty času — detekce strchr(s, ':') a správný sscanf
  • Práce s centisekundami — přesné sčítání a konverze zpět na string
  • Výpočet v strukturách — total_cs se nevytahuje ze souboru, počítá se
  • sprintf() pro formátování čísla do stringu
