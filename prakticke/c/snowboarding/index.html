// C — praktická úloha

Snowboarding

🏂 Paralelní obří slalom 📊 Dvě kola + součet

Zadání

Vytvořte program v jazyce C, který s využitím vzorového datového souboru závodnic paralelního obřího slalomu ve snowboardingu snowboarding.txt vypíše na obrazovku výsledkovou listinu kvalifikačního závodu.

Klíčový rozdíl: Na rozdíl od Jizerské 50 zde máme dvě kola (Modrá dráha, Červená dráha) a celkový čas je jejich součet. Řazení je opět podle celkového času vzestupně.

Přístup k řešení

  1. Struktura — struct pro závodnici: jméno, země, čas kola 1, čas kola 2, celkový čas
  2. Parsování — CSV s pipe separátorem (|)
  3. Součet časů — celkový čas = kolo 1 + kolo 2 (v sekundách)
  4. Řazení — podle celkového času vzestupně
  5. Výpis — tabulka s pořadím, oběma koly a součtem

Kompletní řešení (C)

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define MAX_RACERS 50
#define MAX_NAME 100
#define BUFFER_SIZE 256

typedef struct {
  char name[MAX_NAME];     // Jméno
  char country[3];         // Země
  int time1_sec;           // Kolo 1 v sekundách
  int time2_sec;           // Kolo 2 v sekundách
  int total_sec;           // Součet (pro řazení)
  char time1_str[20];      // Kolo 1 jako string
  char time2_str[20];      // Kolo 2 jako string
  char total_str[20];      // Součet jako string
} Racer;

// Funkce: převést "MM:SS.MS" na sekundy (s přesností na setiny)
int time_to_centiseconds(const char *time_str) {
  int m, s, ms;
  sscanf(time_str, "%d:%d.%d", &m, &s, &ms);
  return m * 6000 + s * 100 + ms;
}

// Funkce: převést setiny zpět na "MM:SS.MS"
void centiseconds_to_string(int centisec, char *str) {
  int m = centisec / 6000;
  int s = (centisec % 6000) / 100;
  int ms = centisec % 100;
  sprintf(str, "%d:%02d.%02d", m, s, ms);
}

// Funkce: výměna obsahu
void swap(Racer *a, Racer *b) {
  Racer temp = *a;
  *a = *b;
  *b = temp;
}

// Funkce: bubble sort podle celkového času
void sort_by_time(Racer *racers, int count) {
  for (int i = 0; i < count - 1; i++) {
    for (int j = 0; j < count - 1 - i; j++) {
      if (racers[j].total_sec > racers[j + 1].total_sec) {
        swap(&racers[j], &racers[j + 1]);
      }
    }
  }
}

// Funkce: čtení a parsování CSV
int read_data(const char *filename, Racer *racers) {
  FILE *file = fopen(filename, "r");
  if (file == NULL) {
    printf("Chyba: Soubor '%s' se nepodařilo otevřít.\n", filename);
    return 0;
  }

  char buffer[BUFFER_SIZE];
  int count = 0;

  // Přeskočit hlavičku
  fgets(buffer, BUFFER_SIZE, file);

  // Čtení řádků
  // Formát: Jméno|Zeme|Modra draha|Cervena draha|Celkovy cas
  while (fgets(buffer, BUFFER_SIZE, file) && count < MAX_RACERS) {
    buffer[strcspn(buffer, "\n")] = 0;

    // Parsování
    char *part = strtok(buffer, "|");
    strcpy(racers[count].name, part);

    part = strtok(NULL, "|");
    while (*part == ' ') part++;
    strcpy(racers[count].country, part);

    // Kolo 1
    part = strtok(NULL, "|");
    while (*part == ' ') part++;
    strcpy(racers[count].time1_str, part);
    racers[count].time1_sec = time_to_centiseconds(part);

    // Kolo 2
    part = strtok(NULL, "|");
    while (*part == ' ') part++;
    strcpy(racers[count].time2_str, part);
    racers[count].time2_sec = time_to_centiseconds(part);

    // Celkový čas (součet obou kol)
    racers[count].total_sec = racers[count].time1_sec +
                              racers[count].time2_sec;
    centiseconds_to_string(racers[count].total_sec,
                           racers[count].total_str);

    count++;
  }

  fclose(file);
  return count;
}

// Funkce: výpis výsledků
void print_results(Racer *racers, int count) {
  printf("\n");
  printf("SNOWBOARD 2024 - KVALIFIKACE paralelní obří slalom ženy\n");
  printf("================================================================\n");
  printf("Por. | Jméno           | Zeme | Modrá dráha  | Červená dráha | Celkem     \n");
  printf("-----+-----------------+------+-------------+---------------+----------\n");

  for (int i = 0; i < count; i++) {
    printf("%3d. | %-15s | %s   | %11s | %13s | %10s\n",
           i + 1,
           racers[i].name,
           racers[i].country,
           racers[i].time1_str,
           racers[i].time2_str,
           racers[i].total_str);
  }
  printf("================================================================\n\n");
}

int main(void) {
  Racer racers[MAX_RACERS];
  int count;

  count = read_data("snowboarding.txt", racers);

  if (count == 0) {
    return 1;
  }

  // Seřadit podle celkového času
  sort_by_time(racers, count);

  // Vypsat výsledky
  print_results(racers, count);

  return 0;
}
Rozdíly oproti Jizerské 50:
• Pipe separator (|) místo středníku
• Dvě kola — oba časy se ukládají do struktury
• Součet časů pro řazení: total_sec = time1_sec + time2_sec
• Čas v setinách sekundy (MM:SS.MS) — pracuje se přesněji
• Výpis obsahuje obě kola i součet

Klíčové koncepty

  • Více polí v struktuře — nejen jeden čas, ale dva, plus součet
  • Počítání součtu — ve struktuře se počítá součet obou kol
  • Pipe separator — CSV s | místo středníku
  • Přesnost času — setiny sekundy, ne celé sekundy
  • Modulárnost — funkce pro konverzi času, swap, třídění, výpis
