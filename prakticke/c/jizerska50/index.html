Jizerská 50

⛷️ Běžecké lyžování 📊 CSV parsing + řazení

Zadání

Vytvořte program v jazyce C, který s využitím vzorového datového souboru přihlášených závodníků v běžeckém lyžování jizerska50.txt vypíše na obrazovku výsledkovou listinu závodu.

Požadavky (17 bodů):

a) Otevření a zavření souboru s kontrolou (2b)
b) Čtení řádků ze souboru (1b)
c) Parsování CSV a mapování do struktur (3b)
d) Práce se strukturami (1b)
e) Zpracování času (2b)
f) Řazení podle času (3b)
g) Funkce pro výměnu obsahu (1b)
h) Výpis výsledkové listiny (2b)
i) Modularita — dělení do funkcí (1b)
j) Symbolické konstanty (1b)

Přístup k řešení

  1. Struktura — definovat struct pro závodníka (číslo, jméno, příjmení, ročník, národnost, čas)
  2. Čtení — otevřít soubor, každý řádek číst do bufferu, parsovat pomocí strtok()
  3. Čas — převést "HH:MM:SS" na sekundy pro srovnání
  4. Řazení — bubble sort nebo qsort() podle času vzestupně
  5. Výpis — formátovaný výstup s pořadím, seznamem, časy

Kompletní řešení (C)

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define MAX_SKIERS 100
#define MAX_NAME 50
#define BUFFER_SIZE 256

// Struktura pro závodníka
typedef struct {
  int number;           // číslo
  char name[MAX_NAME];  // jméno
  char surname[MAX_NAME]; // příjmení
  int year;             // ročník
  char country[3];      // státní kód (NOR, SWE, ...)
  int time_sec;         // čas v sekundách (pro srovnání)
  char time_str[20];    // čas jako string (HH:MM:SS)
} Skier;

// Funkce: převést "HH:MM:SS" na sekundy
int time_to_seconds(const char *time_str) {
  int h, m, s;
  sscanf(time_str, "%d:%d:%d", &h, &m, &s);
  return h * 3600 + m * 60 + s;
}

// Funkce: výměna obsahu dvou struktur
void swap(Skier *a, Skier *b) {
  Skier temp = *a;
  *a = *b;
  *b = temp;
}

// Funkce: bubble sort podle času
void sort_by_time(Skier *skiers, int count) {
  for (int i = 0; i < count - 1; i++) {
    for (int j = 0; j < count - 1 - i; j++) {
      if (skiers[j].time_sec > skiers[j + 1].time_sec) {
        swap(&skiers[j], &skiers[j + 1]);
      }
    }
  }
}

// Funkce: čtení a parsování CSV souboru
int read_data(const char *filename, Skier *skiers) {
  FILE *file = fopen(filename, "r");
  if (file == NULL) {
    printf("Chyba: Soubor '%s' se nepodařilo otevřít.\n", filename);
    return 0;
  }

  char buffer[BUFFER_SIZE];
  int count = 0;

  // Přeskočit hlavičku
  fgets(buffer, BUFFER_SIZE, file);

  // Čtení řádků
  while (fgets(buffer, BUFFER_SIZE, file) && count < MAX_SKIERS) {
    // Odstranit newline
    buffer[strcspn(buffer, "\n")] = 0;

    // Parsování: "číslo;příjmení, jméno (ročník);národnost;čas"
    char *part = strtok(buffer, ";");
    skiers[count].number = atoi(part);

    // Jméno a příjmení (formát: "Surname, Firstname (Year)")
    part = strtok(NULL, ";");
    char temp[MAX_NAME * 2];
    strcpy(temp, part);

    // Rozdělit: příjmení,jméno,ročník
    char *name_part = strtok(temp, ",");
    strcpy(skiers[count].surname, name_part);

    // Jméno (za ", ")
    name_part = strtok(NULL, "(");
    while (*name_part == ' ') name_part++;
    char *paren = strchr(name_part, ' ');
    if (paren) *paren = 0;
    strcpy(skiers[count].name, name_part);

    // Ročník
    part = strtok(NULL, ")");
    skiers[count].year = atoi(part);

    // Národnost
    part = strtok(NULL, ";");
    while (*part == ' ') part++;
    strcpy(skiers[count].country, part);

    // Čas
    part = strtok(NULL, ";");
    while (*part == ' ') part++;
    strcpy(skiers[count].time_str, part);
    skiers[count].time_sec = time_to_seconds(part);

    count++;
  }

  fclose(file);
  return count;
}

// Funkce: výpis výsledkové listiny
void print_results(Skier *skiers, int count) {
  printf("\n");
  printf("VÝSLEDKOVÁ LISTINA - JIZERSKÁ 50\n");
  printf("=====================================================\n");
  printf("Por. | Číslo | Příjmení         | Jméno    | Čas      \n");
  printf("-----+-------+------------------+----------+----------\n");

  for (int i = 0; i < count; i++) {
    printf("%3d. | %5d | %-16s | %-8s | %s\n",
           i + 1,
           skiers[i].number,
           skiers[i].surname,
           skiers[i].name,
           skiers[i].time_str);
  }
  printf("=====================================================\n\n");
}

// MAIN
int main(void) {
  Skier skiers[MAX_SKIERS];
  int count;

  // Načíst data z souboru
  count = read_data("jizerska50.txt", skiers);

  if (count == 0) {
    return 1;
  }

  // Seřadit podle času
  sort_by_time(skiers, count);

  // Vypsat výsledky
  print_results(skiers, count);

  return 0;
}
Klíčové body kódu:
• strtok() pro parsování CSV po středníku
• time_to_seconds() konvertuje čas na sekundy pro srovnání
• swap() funkce pro výměnu prvků (bubble sort)
• sort_by_time() seřadí podle času vzestupně
• Všechna data jsou v jedné struktuře, kód je modulární
