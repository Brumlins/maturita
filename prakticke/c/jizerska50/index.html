Jizerská 50

⛷️ Běžecké lyžování 📄 CSV se středníkem ⏱ Čas HH:MM:SS

Zadání

Vytvořte program v jazyce C, který načte soubor jizerska50.txt s výsledky závodu v běžeckém lyžování, seřadí závodníky podle dosažených časů a vypíše výsledkovou listinu.

Formát souboru: CSV oddělené středníkem. Jméno je ve složitějším formátu: Příjmení, Jméno (Ročník) — vše v jednom sloupci.

Ukázka dat (jizerska50.txt)

st. cislo;prijmeni, jmeno (rocnik);narodnost;cas
22;Persson, Emil (1995);SWE;2:02:27
23;Kardin, Oskar (1992);SWE;2:02:38
19;Hoelgaard, Stian (1991);NOR;2:03:34
40;Srail, Jan (1990);CZE;2:07:53
...

Kompletní řešení

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define MAX_SKIERS  100
#define MAX_NAME    50
#define BUFFER_SIZE 256

typedef struct {
    int  number;
    char name[MAX_NAME];
    char surname[MAX_NAME];
    int  year;
    char country[4];   /* 3 znaky + '\0' */
    int  time_sec;
    char time_str[20];
} Skier;

/* "2:02:27" → sekundy */
int time_to_seconds(const char *s) {
    int h, m, sec;
    sscanf(s, "%d:%d:%d", &h, &m, &sec);
    return h * 3600 + m * 60 + sec;
}

void swap(Skier *a, Skier *b) {
    Skier tmp = *a; *a = *b; *b = tmp;
}

void sort_by_time(Skier *arr, int n) {
    for (int i = 0; i < n - 1; i++)
        for (int j = 0; j < n - 1 - i; j++)
            if (arr[j].time_sec > arr[j+1].time_sec)
                swap(&arr[j], &arr[j+1]);
}

int read_data(const char *filename, Skier *arr) {
    FILE *f = fopen(filename, "r");
    if (!f) { printf("Chyba: nelze otevrit '%s'\n", filename); return 0; }

    char buf[BUFFER_SIZE];
    int n = 0;

    fgets(buf, sizeof(buf), f);  /* přeskočit hlavičku */

    /* Formát: cislo;Prijmeni, Jmeno (Rocnik);Narodnost;Cas */
    while (fgets(buf, sizeof(buf), f) && n < MAX_SKIERS) {
        buf[strcspn(buf, "\n")] = 0;

        /* číslo */
        char *p = strtok(buf, ";");
        arr[n].number = atoi(p);

        /* "Hoelgaard, Stian (1991)" */
        p = strtok(NULL, ";");
        char tmp[MAX_NAME * 2];
        strcpy(tmp, p);

        char *tok = strtok(tmp, ",");
        strcpy(arr[n].surname, tok);          /* příjmení */

        tok = strtok(NULL, "(");
        while (*tok == ' ') tok++;
        char *sp = strchr(tok, ' ');
        if (sp) *sp = 0;
        strcpy(arr[n].name, tok);             /* jméno */

        tok = strtok(NULL, ")");
        arr[n].year = atoi(tok);              /* ročník */

        /* národnost */
        p = strtok(NULL, ";");
        while (*p == ' ') p++;
        strcpy(arr[n].country, p);

        /* čas */
        p = strtok(NULL, ";");
        while (*p == ' ') p++;
        strcpy(arr[n].time_str, p);
        arr[n].time_sec = time_to_seconds(p);

        n++;
    }
    fclose(f);
    return n;
}

void print_results(Skier *arr, int n) {
    printf("\nVYSLEDKOVA LISTINA - JIZERSKA 50\n");
    printf("=====================================================\n");
    printf("Por. | Cislo | Prijmeni         | Jmeno    | Cas\n");
    printf("-----+-------+------------------+----------+----------\n");
    for (int i = 0; i < n; i++)
        printf("%3d. | %5d | %-16s | %-8s | %s\n",
               i+1, arr[i].number, arr[i].surname,
               arr[i].name, arr[i].time_str);
    printf("=====================================================\n\n");
}

int main(void) {
    Skier skiers[MAX_SKIERS];
    int n = read_data("jizerska50.txt", skiers);
    if (n == 0) return 1;
    sort_by_time(skiers, n);
    print_results(skiers, n);
    return 0;
}
Klíčové body:
• strtok() rozděluje řádek po středníku
• Jméno je v jednom poli jako "Příjmení, Jméno (Ročník)" — rozebíráme znovu pomocí strtok() se znaky , a ()
• country[4] — 3 znaky ("NOR") + ukončovací '\0' = musí být velikost 4
• Bubble sort porovnává time_sec (celá čísla), vypisuje time_str (původní string)

Co se naučíš

  • Otevření souboru — fopen() s kontrolou if (!f)
  • Čtení řádků — fgets() do bufferu + oříznutí \n přes strcspn()
  • Parsování CSV — strtok() nejprve podle ;, pak uvnitř pole znovu
  • Struktury — typedef struct, pole struktur
  • Konverze času — sscanf() pro HH:MM:SS → sekundy
  • Bubble sort — vnořené for cykly + funkce swap()
  • Formátovaný výstup — printf() s %-16s, %5d atd.
