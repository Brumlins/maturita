Základy — vytvoření a spuštění skriptu

Vytvoření + spuštění 
# 1. Vytvořit skript:
$ mcedit skript.sh

# 2. Nastavit spustitelný:
$ chmod +x skript.sh

# 3. Spustit:
$ ./skript.sh
Struktura bash skriptu 
#!/bin/bash
# Shebang — říká systému, čím skript spustit

# Proměnná:
jmeno="Alice"
echo "Ahoj $jmeno"

# Výstup příkazu do proměnné:
datum=`date +%Y-%m-%d`
# nebo modernější zápis:
datum=$(date +%Y-%m-%d)

Proměnné, pole, parsování textu

Pole a parsování řetězce 
#!/bin/bash
prom="moje prvni sluzba"

# Rozdělení na pole (podle mezer):
pole=($prom)
echo ${pole[0]}       # moje
echo ${pole[1]}       # prvni
echo ${#pole[@]}     # počet prvků pole

# Smazání podřetězce:
echo $prom | tr -d 'prvni'

# Záměna podřetězce:
echo $prom | tr 'prvni' 'druha'
Čtení souboru po řádcích + split podle oddělovače 
#!/bin/bash
# Soubor users.txt: jmeno;heslo;skupina
filename='users.txt'

cat $filename | while read line
do
  # Rozdělení podle středníku:
  bar=(`echo $line | tr ';' ' '`)
  echo "Jméno: ${bar[0]}, Heslo: ${bar[1]}, Skupina: ${bar[2]}"
done

if / else — kontrola existence adresáře 
#!/bin/bash
if test -e "/tmp/zaloha"; then
  echo "adresář existuje"
else
  echo "adresář neexistuje, bude vytvořen"
  mkdir /tmp/zaloha
fi

# Kontrola návratové hodnoty příkazu ($?):

  echo "OK"

  echo "CHYBA"
fi
for cyklus přes pole 
#!/bin/bash
# Projít skupiny uživatele a hledat 'adm':
skupiny=`groups`
pole=($skupiny)

for skupina in ${pole[@]}; do
  if [ $skupina = 'adm' ]; then
    echo "uživatel je ve skupině adm"
  fi
done

Vytváření uživatelů ze souboru

create-users.sh — čte users.txt 
#!/bin/bash
# users.txt formát: jmeno;heslo;skupina1;skupina2

filename='users.txt'
cat $filename | while read line
do
  bar=(`echo $line | tr ';' ' '`)
  jmeno=${bar[0]}
  heslo=${bar[1]}
  skupina=${bar[2]}

  # Vytvořit skupinu (pokud neexistuje):
  groupadd -f $skupina

  # Vytvořit uživatele ve skupině:
  useradd -m -g $skupina $jmeno

  # Nastavit heslo:
  echo "${jmeno}:${heslo}" | chpasswd

  echo "Vytvořen uživatel: $jmeno"
done

Monitorovací skripty

Kontrola dostupnosti serveru (ping) 
#!/bin/bash
host="192.168.0.1"

ping -s 1450 -c 10 $host > /tmp/ping_out.txt
ztrata=`cat /tmp/ping_out.txt | grep 'packet loss' | awk '{print $6}' | tr -d '%'`

if [ $ztrata -gt 0 ]; then
  echo "PROBLEM: $host nedostupný, ztráta $ztrata%"
else
  echo "OK: $host dostupný"
fi

#!/bin/bash

Záloha MySQL databáze 
#!/bin/bash
datum=`date +%d%m%y`

mysqldump -u root -p"heslo" --all-databases | gzip -c > /tmp/${datum}-mysql.gz

  logger "MySQL záloha OK"

  logger "MySQL záloha SELHALA"

Počet paketů na port 80 přes iptables 
# Přidat pravidlo pro sledování:
$ iptables -A OUTPUT -p tcp --dport 80 -j ACCEPT

$ iptables -L -v | grep 'tcp dpt:http' | awk '{print $1}'

$ iptables -F
