Bash skripty — maturitní příprava
📄 blog 📄 příklady
1 — Základní struktura skriptu
Vytvoření a spuštění
#!/bin/bash
# Nastavit spustitelný a spustit:
$ chmod +x skript.sh
$ ./skript.sh
2 — Proměnné
Proměnné a vstup
#!/bin/bash
jmeno="Alice"
datum=`date +%d%m%y`
echo "Ahoj $jmeno, dnes je $datum"

# Čtení vstupu od uživatele:
read -p "Zadej jméno: " jmeno
3 — Podmínky (if)
if / else — kontrola návratového kódu
#!/bin/bash
tar cvfz /tmp/zaloha.tar.gz /etc

if [ "$?" -eq "0" ]; then
  echo "Záloha OK"
else
  echo "Záloha SELHALA"
fi

# $? = návratový kód posledního příkazu (0 = úspěch)
4 — Cyklus for
for — iterace přes soubory / seznam
#!/bin/bash
# Iterace přes soubory:
for soubor in /etc/*.conf; do
  echo "Soubor: $soubor"
done

# Číselný rozsah:
for i in {1..5}; do
  echo "Řádek $i"
done
5 — Zpracování souboru řádek po řádku
while read — načtení ze souboru
#!/bin/bash
# soubor.txt obsahuje: jmeno:heslo (každý uživatel na řádku)
while IFS=':' read jmeno heslo; do
  useradd -m $jmeno
  echo "$jmeno:$heslo" | chpasswd
  echo "Vytvořen uživatel: $jmeno"
done < soubor.txt
6 — Zálohovací skript s logováním
backup.sh — záloha /etc s logger
#!/bin/bash
datum=`date +%d%m%y`

tar cvfz /tmp/${datum}zaloha.tar.gz /etc

  logger "záloha OK"
else

// CRON — automatické spouštění Přidat do cronu: crontab -e
Denně ve 2:00: 0 2 * * * /root/backup.sh
Instalace WordPress na Ubuntu 18.04+
📄 blog
1 — Instalace LAMP stacku
Apache + MySQL + PHP
# Aktualizace a instalace:
$ apt-get update
$ apt-get install apache2 mysql-server php php-mysql php-curl php-gd php-mbstring php-xml

# Zabezpečení MySQL:
$ mysql_secure_installation
2 — Vytvoření databáze pro WordPress
MySQL — databáze a uživatel
$ mysql -u root -p

-- V MySQL konzoli:
CREATE DATABASE wordpress;
CREATE USER 'wpuser'@'localhost' IDENTIFIED BY 'heslo';
GRANT ALL PRIVILEGES ON wordpress.* TO 'wpuser'@'localhost';
FLUSH PRIVILEGES;
EXIT;
3 — Stažení a instalace WordPress
Stažení a rozbalení WordPress
$ cd /var/www/html
$ wget https://wordpress.org/latest.tar.gz
$ tar xvfz latest.tar.gz
$ chown -R www-data:www-data /var/www/html/wordpress
4 — Konfigurace wp-config.php
wp-config.php — databázové nastavení
$ cp /var/www/html/wordpress/wp-config-sample.php /var/www/html/wordpress/wp-config.php
$ nano /var/www/html/wordpress/wp-config.php

# Upravit tyto řádky:
define('DB_NAME',     'wordpress');
define('DB_USER',     'wpuser');
define('DB_PASSWORD', 'heslo');
define('DB_HOST',     'localhost');
5 — Dokončení v prohlížeči
// Otevři v prohlížeči: http://localhost/wordpress

Samba server (sdílení adresářů)
📄 blog
1 — Instalace
Debian/Ubuntu
$ apt-get install samba
$ samba --version
2 — Příprava adresáře a uživatele
Adresář + systémový uživatel + Samba heslo
# Vytvoření sdíleného adresáře:
$ mkdir /data
$ chown student:users /data

# Přidat uživatele do skupiny users:
$ usermod -a -G users student

# Nastavit Samba heslo (odlišné od systémového!):
$ smbpasswd -a student

# Kontrola Samba uživatelů:
$ pdbedit -L
3 — Konfigurace smb.conf
/etc/samba/smb.conf — přidat na konec
[accounting]
  comment        = Sdileny adresar
  path           = /data
  valid users    = @users
  writable       = yes
  force group    = users
  create mode    = 0660
  directory mode = 0770
4 — Restart a ověření
Testování konfigurace a restart
# Ověřit syntaxi:
$ testparm

# Restart:
$ systemctl restart smbd nmbd
// Připojení z Windows Win+R → \\192.168.0.10\accounting → zadej Samba uživatele a heslo
NFS server
📄 blog ↗ DigitalOcean
1 — Server: instalace a adresář
Server — NFS kernel server
$ apt-get install nfs-kernel-server

# Vytvoření sdíleného adresáře:
$ mkdir /data
$ chmod 777 /data
# nebo:
$ chown nobody:nogroup /data
2 — /etc/exports
/etc/exports — přidat na konec
# Přístup všem (*):
/data *(rw,no_root_squash,no_subtree_check,sync)

# Přístup konkrétní IP:
/data 192.168.0.10(rw,sync,no_subtree_check)

# Přístup celé síti:
/data 192.168.0.0/24(rw,sync,no_subtree_check)
Aplikace změn + restart
$ exportfs -ra
$ systemctl restart nfs-kernel-server
# Ověření aktivních sdílení:
$ exportfs -v
3 — Klient: připojení
Klient — připojení NFS sdílení
$ apt-get install nfs-common
$ mkdir /media/nfs
$ mount -t nfs 192.168.0.10:/data /media/nfs

# Zobrazit dostupná sdílení na serveru:
$ showmount -e 192.168.0.10
4 — Trvalé připojení (/etc/fstab)
/etc/fstab — přidat na konec
192.168.0.10:/data  /media/nfs  nfs  defaults,_netdev  0 0

# Test (připojí vše z fstab):
$ mount -a
SSH a práce s klíči
📄 blog
1 — Instalace SSH serveru
OpenSSH server
$ apt-get install openssh-server
$ systemctl status ssh
$ systemctl enable ssh
2 — Generování SSH klíče
Generování RSA klíče (na klientovi)
# Generování klíče (RSA 4096 bit):
$ ssh-keygen -t rsa -b 4096
# Výsledek: ~/.ssh/id_rsa (soukromý) + ~/.ssh/id_rsa.pub (veřejný)
3 — Zkopírování klíče na server
ssh-copy-id nebo ručně
# Automaticky (doporučeno):
$ ssh-copy-id uzivatel@server

# Ručně — přidat obsah id_rsa.pub do souboru na serveru:
$ cat ~/.ssh/id_rsa.pub >> ~/.ssh/authorized_keys
$ chmod 600 ~/.ssh/authorized_keys
4 — Zákaz přihlášení heslem
/etc/ssh/sshd_config
# Upravit řádky v /etc/ssh/sshd_config:
PasswordAuthentication no
PermitRootLogin        no
PubkeyAuthentication   yes

$ systemctl restart ssh
5 — SCP — kopírování souborů
SCP
# Nahrát soubor na server:
$ scp soubor.txt uzivatel@server:/home/uzivatel/

# Stáhnout ze serveru:
$ scp uzivatel@server:/etc/hosts ./hosts_zaloha
Apache2 server — userdir konfigurace
📄 blog
1 — Instalace
Instalace + ověření
$ apt-get install apache2
$ systemctl status apache2
$ systemctl enable apache2
# Otevři: http://localhost
// DŮLEŽITÉ CESTY Web root: /var/www/html/  ·  Konfigurace: /etc/apache2/  ·  Logy: /var/log/apache2/
2 — Správa modulů
Aktivace / deaktivace modulů
# Aktivace:
$ a2enmod userdir
$ a2enmod rewrite

# Deaktivace:
$ a2dismod userdir

# Reload:
$ systemctl reload apache2
3 — Userdir modul
~/public_html → http://server/~uzivatel/
# 1. Aktivovat modul:
$ a2enmod userdir
$ service apache2 restart

# 2. Vytvořit adresář a nastavit práva:
$ mkdir ~/public_html
$ chmod 755 ~/public_html

# 3. Testovací stránka:
$ echo "<h1>Moje stránka</h1>" > ~/public_html/index.html

# 4. Test: http://localhost/~username/
// PHP V USERDIR PHP v userdir bývá vypnuté. Oprav v /etc/apache2/mods-available/php8.x.conf — zakomentuj sekci php_admin_value engine Off uvnitř bloku mod_userdir.c.
4 — Virtual Hosts
/etc/apache2/sites-available/mujweb.conf
<VirtualHost *:80>
  ServerName   mujweb.localhost
  DocumentRoot /var/www/mujweb
  ErrorLog     ${APACHE_LOG_DIR}/mujweb-error.log
</VirtualHost>
Aktivace virtual hostu
# Přidat do /etc/hosts:
127.0.0.1    mujweb.localhost

$ mkdir /var/www/mujweb
$ a2ensite mujweb.conf
$ systemctl reload apache2
5 — phpMyAdmin
Instalace phpMyAdmin
$ apt-get install phpmyadmin php-mbstring

# Pokud nefunguje http://localhost/phpmyadmin:
$ ln -s /usr/share/phpmyadmin /var/www/html/phpmyadmin
$ systemctl reload apache2
Zálohování a obnova adresáře (tar)
📄 blog
1 — Parametry
ParametrFunkce
ccreate — vytvoří nový archiv
xextract — rozbalí archiv
vverbose — vypisuje zpracovávané soubory
ffile — specifikuje jméno archivu (vždy nutné)
zgzip komprese (.tar.gz)
jbzip2 komprese (.tar.bz2)
Ccílový adresář pro rozbalení
tlist — zobrazí obsah bez rozbalení
2 — Vytvoření zálohy
Záloha adresáře
# Syntaxe: tar cvfz [výstupní soubor] [co zálohovat]
$ tar cvfz /tmp/zaloha-etc.tar.gz /etc

# S časovým razítkem:
$ tar cvfz /tmp/zaloha-$(date +%Y%m%d).tar.gz /etc

# Zobrazení obsahu bez rozbalení:
$ tar tvfz /tmp/zaloha-etc.tar.gz
3 — Rozbalení zálohy
Rozbalení archivu
# Syntaxe: tar xvfz [soubor] -C [kam rozbalit]
$ tar xvfz /tmp/zaloha-etc.tar.gz -C /tmp/obnova/

# Jen konkrétní soubor z archivu:
$ tar xvfz zaloha.tar.gz etc/hostname

1 — mysqldump — záloha databáze
Záloha jedné / všech databází
# Záloha jedné databáze:
$ mysqldump -u root -p jmeno_databaze > /tmp/zaloha.sql

# Záloha všech databází:
$ mysqldump -u root -p --all-databases > /tmp/vsechny.sql

# Komprimovaná záloha (gzip):
$ mysqldump -u root -p"heslo" --all-databases | gzip -c > /tmp/zaloha.gz
2 — Obnova ze zálohy
Obnovení databáze
# Obnova ze SQL souboru:
$ mysql -u root -p jmeno_databaze < /tmp/zaloha.sql

# Obnova z komprimované zálohy:
$ gunzip -c /tmp/zaloha.gz | mysql -u root -p

# Vytvoření prázdné DB před obnovením (pokud neexistuje):
$ mysql -u root -p -e "CREATE DATABASE jmeno_databaze;"
3 — Automatický zálohovací skript
mysql-backup.sh
#!/bin/bash
datum=`date +%d%m%y`

mysqldump -u root -p"heslo" --all-databases | gzip -c > /tmp/${datum}-mysql.gz

if [ "$?" -eq "0" ]; then
  logger "MySQL záloha OK"
else
  logger "MySQL záloha SELHALA"
fi
4 — Spuštění a nastavení práv
Spuštění skriptu
$ chmod +x mysql-backup.sh
$ ./mysql-backup.sh

# Cron — denně ve 3:00:
$ crontab -e
0 3 * * * /root/mysql-backup.sh
// mysqldump vs fyzická záloha mysqldump vytvoří SQL soubor (přenositelný, čitelný). Fyzická záloha (/var/lib/mysql/) je rychlejší ale závislá na verzi MySQL.

Soubory a adresáře
Základní práce se soubory
ls -la              # výpis adresáře
cd /cesta           # přechod do adresáře
mkdir adresar      # vytvoření adresáře
cp zdroj cíl       # kopírování
mv zdroj cíl       # přesun / přejmenování
rm -rf adresar    # smazání rekurzivně
find /etc -name "*.conf"   # hledání souborů
grep -r "text" /etc       # hledání textu v souborech
Oprávnění
chmod / chown
chmod 755 soubor           # rwxr-xr-x
chmod 644 soubor           # rw-r--r--
chmod -R 755 /var/www     # rekurzivně
chown user:skupina soubor  # změna vlastníka
chown -R www-data:www-data /var/www
Uživatelé a skupiny
useradd / groupadd
useradd -m jmeno         # vytvoří uživatele + home
passwd jmeno             # nastaví heslo
userdel -r jmeno        # smaže uživatele + home
groupadd skupina         # vytvoří skupinu
usermod -a -G skupina jmeno  # přidat do skupiny
id jmeno                 # UID, GID, skupiny
