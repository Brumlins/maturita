const int MIN_ROK = 1970;
const int MAX_ROK = 2050;

class Datum {
public:
    Datum();
    Datum(int y, int m, int d);
    Datum(const Datum&);

    int  get_year() const;
    int  get_month() const;
    int  get_day() const;
    bool set_year(int y);
    bool set_month(int m);
    bool set_day(int d);
    bool set_date(int y, int m, int d);

    int  days_from_1970() const;
    int  day_in_week() const;
    bool set_from_days(int days);
    bool change_date(int days);

    Datum& operator=(const Datum& diff);
    bool   operator==(const Datum& diff) const;
    friend std::ostream& operator<<(std::ostream& os, const Datum& d);

private:
    int y, m, d;

    static bool je_valid(int y, int m, int d);
    bool        je_prestupny() const;
    static bool je_prestupny(int y);
    static int  pocet_dni(int m, int y);
    void        system_datum();
};
#include "datum.h"
#include <ctime>

Datum::Datum() { system_datum(); }

void Datum::system_datum() {
    time_t ts = time(&ts);
    struct tm* dt = localtime(&ts);
    y = dt->tm_year + 1900;
    m = dt->tm_mon  + 1;
    d = dt->tm_mday;
}

Datum::Datum(int y, int m, int d) {
    if (je_valid(y, m, d)) { this->y=y; this->m=m; this->d=d; }
    else system_datum();
}

Datum::Datum(const Datum& o) { y=o.y; m=o.m; d=o.d; }

int Datum::get_year()  const { return y; }
int Datum::get_month() const { return m; }
int Datum::get_day()   const { return d; }

bool Datum::set_year(int y) {
    if (je_valid(y, this->m, this->d)) { this->y=y; return true; }
    return false;
}

bool Datum::set_month(int m) {
    if (je_valid(this->y, m, this->d)) { this->m=m; return true; }
    return false;
}

bool Datum::set_day(int d) {
    if (je_valid(this->y, this->m, d)) { this->d=d; return true; }
    return false;
}

bool Datum::set_date(int y, int m, int d) {
    if (je_valid(y, m, d)) { this->y=y; this->m=m; this->d=d; return true; }
    return false;
}

int Datum::days_from_1970() const {
    int total = 0;
    for (int i = MIN_ROK; i < y; ++i)          // celé roky
        total += je_prestupny(i) ? 366 : 365;
    for (int i = 1; i < m; ++i)               // celé měsíce před aktuálním
        total += pocet_dni(i, y);
    total += d;                                 // dny v aktuálním měsíci
    return total;
}

int Datum::day_in_week() const {
    return (days_from_1970() + 2) % 7 + 1;    // pondělí=1, 1.1.1970 byl čtvrtek=4
}

bool Datum::set_from_days(int days) {
    if (days <= 0) return false;
    int ny = MIN_ROK;
    while (days > (je_prestupny(ny) ? 366 : 365)) {
        days -= je_prestupny(ny) ? 366 : 365;
        if (++ny > MAX_ROK) return false;
    }
    int nm = 1;
    while (days > pocet_dni(nm, ny)) {
        days -= pocet_dni(nm, ny);
        ++nm;
    }
    y = ny; m = nm; d = days;
    return true;
}

bool Datum::change_date(int days) {
    return set_from_days(days_from_1970() + days);
}

Datum& Datum::operator=(const Datum& diff) {
    y=diff.y; m=diff.m; d=diff.d;
    return *this;
}

bool Datum::operator==(const Datum& diff) const {
    return y==diff.y && m==diff.m && d==diff.d;
}

bool Datum::je_valid(int y, int m, int d) {
    if (y < MIN_ROK || y > MAX_ROK) return false;
    if (m < 1 || m > 12)            return false;
    if (d < 1 || d > pocet_dni(m,y)) return false;
    return true;
}

bool Datum::je_prestupny(int y) {
    return y%4==0 && (y%400==0 || y%100!=0);
}

bool Datum::je_prestupny() const { return je_prestupny(y); }

int Datum::pocet_dni(int m, int y) {
    int mes[] = {31,28,31,30,31,30,31,31,30,31,30,31};
    if (je_prestupny(y)) mes[1] = 29;
    return mes[m-1];
}

std::ostream& operator<<(std::ostream& os, const Datum& d) {
    os << d.d << ". " << d.m << ". " << d.y << ", ";
    os << "pocet dni od 1. 1. " << MIN_ROK << " je " << d.days_from_1970() << ", ";
    os << "den v tydnu je " << d.day_in_week() << ".";
    return os;
}
Klíčové C++ vzory v tomto příkladu
static member funkce — volatelné bez objektu
Funkce jako je_valid(), je_prestupny(int) a pocet_dni() jsou static — nepotřebují this, pracují jen s parametry. Voláš je jako Datum::pocet_dni(m, y) nebo přímo uvnitř třídy jako pocet_dni(m, y).
Přetěžování konstruktoru — tři varianty
Výchozí konstruktor volá system_datum(). Parametrický nejprve validuje přes je_valid(), při neplatném vstupu fallbackuje na system_datum(). Kopírovací prostě zkopíruje tři int členy.
friend operator<< — přístup k private bez dědičnosti
Deklaruješ v .h: friend std::ostream& operator<<(std::ostream&, const Datum&);. Implementuješ v .cpp mimo třídu (bez Datum::). Protože je friend, má přístup k d.d, d.m, d.y i když jsou private.
Přestupný rok — vzorec
y%4==0 && (y%400==0 || y%100!=0) — dělitelný 4, ale staletí nejsou přestupná (1900), kromě těch dělitelných 400 (2000).
