Algoritmus validace rodného čísla
Jak funguje kontrolní číslice:
RC má 10 číslic: RRMMDD/XXXk (nebo bez lomítka: RRMMDDXXXk)
1. Číslo z prvních 9 číslic % 11 = zbytek
2. Zbytek == poslední číslice → platné
3. Zbytek == 10 a poslední == 0 → taky platné (speciální případ)

Parsování data z RC:
• Rok: substr(0,2) → r > 54 → 1900+r, jinak 2000+r
• Měsíc: substr(2,2) → m > 50 → žena (m-50), jinak muž
• Den: substr(4,2) → přímo den

#ifndef RODNE_H
#define RODNE_H
#include <string>
#include <ostream>
using std::ostream;
using std::string;

const string default_rc = "5451010004";

class rodne {
public:
    rodne();
    ~rodne();
    rodne(string);
    rodne(const rodne&);
    string get_rodne() const;
    bool   set_rodne(string rodne);
    int    den() const;
    int    mesic() const;
    int    rok() const;
    int    vek() const;
    bool   pohlavi() const;
    friend ostream& operator<<(ostream&, const rodne&);

private:
    string rodneCislo;
    static bool validace(string rodne);
};

#endif
#include "rodne.h"
#include <ctime>

rodne::rodne()           { rodneCislo = default_rc; }
rodne::~rodne()          {}
rodne::rodne(string rc)  { if (!set_rodne(rc)) rodneCislo = default_rc; }
rodne::rodne(const rodne& rc) { rodneCislo = rc.get_rodne(); }

string rodne::get_rodne() const { return rodneCislo; }

bool rodne::set_rodne(string rc) {
    if (validace(rc)) { rodneCislo = rc; return true; }
    return false;
}

int rodne::den() const {
    return std::stoi(rodneCislo.substr(4, 2));
}

int rodne::mesic() const {
    int m = std::stoi(rodneCislo.substr(2, 2));
    return m > 50 ? m - 50 : m;         // ženy mají měsíc +50
}

int rodne::rok() const {
    int r = std::stoi(rodneCislo.substr(0, 2));
    return r > 54 ? 1900 + r : 2000 + r;
}

int rodne::vek() const {
    time_t raw; struct tm* ti;
    time(&raw); ti = localtime(&raw);
    int s_rok   = ti->tm_year + 1900;
    int s_mesic = ti->tm_mon  + 1;
    int s_den   = ti->tm_mday;
    int vek = s_rok - this->rok();
    if (s_mesic < this->mesic() ||
       (s_mesic == this->mesic() && s_den < this->den()))
        --vek;                               // narozeniny ještě nebyly letos
    return vek;
}

bool rodne::pohlavi() const {
    return std::stoi(rodneCislo.substr(2, 2)) > 50;  // true = žena
}

bool rodne::validace(string rc) {
    if (rc.length() != 10) return false;
    for (int i = 0; i < (int)rc.length(); ++i)
        if (rc[i] < '0' || rc[i] > '9') return false;
    int zbytek = std::stoi(rc.substr(0,9)) % 11;
    int posl   = std::stoi(rc.substr(9,1));
    if (zbytek == posl) return true;
    if (zbytek == 10 && posl == 0) return true;
    return false;
}

ostream& operator<<(ostream& os, const rodne& r) {
    os << "Rodne cislo: "  << r.rodneCislo << std::endl;
    os << "Narozen:     "  << r.den() << ". " << r.mesic() << ". " << r.rok() << std::endl;
    os << "Pohlavi:     "  << (r.pohlavi() ? "Zena" : "Muz") << std::endl;
    os << "Vek:         "  << r.vek() << std::endl;
    return os;
}
Klíčové vzory v tomto příkladu
string::substr(pos, len) + stoi()
RC je uloženo jako string. Extrahujeme části: rodneCislo.substr(0,2) = první 2 znaky (rok), substr(2,2) = měsíc, substr(4,2) = den. Pak std::stoi() převede string na int.
static validace — bez objektu, jen čistá funkce
validace() je static — nevyžaduje objekt, pracuje jen s parametrem string rc. Voláme ji v konstruktoru: set_rodne(rc) → validace(rc). Výhodou je, že ji lze volat ještě před vytvořením objektu.
Věk — systémové datum pomocí time.h
time_t raw; time(&raw); struct tm* ti = localtime(&raw); — ti→tm_year je roky od 1900, proto +1900. ti→tm_mon je 0-11, proto +1. Věk = s_rok - rok_narozeni, pak korekce: pokud narozeniny ještě nebyly (měsíc nebo den je menší), odečti 1.
