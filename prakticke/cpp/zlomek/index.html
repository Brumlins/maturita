Klíčové algoritmy
Sčítání zlomků (operator+=):
1. Najdi LCM (nejmenší společný násobek) jmenovatelů: lcm(a, b) z <numeric>
2. Převeď oba čitatele na společného jmenovatele: lcm/jmen * cit
3. Sečti čitatele, jmenovatel je lcm
Příklad: 3/4 + 1/6 → lcm(4,6)=12 → 9/12 + 2/12 = 11/12
Krácení (reduce) pomocí GCD:
gcd(numerator, denominator) z <numeric> (C++17)
Dělíme oba GCD: 6/10 → gcd(6,10)=2 → 3/5
Pokud GCD=1, krácení není možné.

class fraction {
public:
    fraction();
    fraction(int numerator, int denominator);
    fraction(const fraction&);
    ~fraction();

    int   getNumerator() const;
    int   getDenominator() const;
    float toDec() const;
    bool  flip();
    bool  reduce();
    bool  setFraction(int num, int den);

    fraction& operator=(const fraction&);
    fraction& operator+=(const fraction&);
    fraction  operator+(const fraction&) const;

private:
    int numerator;
    int denominator;
    void swap(int& num, int& den);
};

ostream& operator<<(ostream&, const fraction&);

#endif
#include "fraction.h"
#include <numeric>        // std::gcd, std::lcm

fraction::fraction() : numerator(1), denominator(1) {}

fraction::fraction(int num, int den) {
    if (!setFraction(num, den)) setFraction(1, 1);
}

fraction::fraction(const fraction& a) {
    setFraction(a.getNumerator(), a.getDenominator());
}

fraction::~fraction() {}

int   fraction::getNumerator()   const { return numerator;   }
int   fraction::getDenominator() const { return denominator; }
float fraction::toDec() const {
    return (float)numerator / (float)denominator;
}

bool fraction::flip() {
    if (numerator == 0) return false;
    swap(numerator, denominator);
    return true;
}

bool fraction::reduce() {
    int nsd = std::gcd(numerator, denominator);
    if (nsd == 1) return false;
    numerator   /= nsd;
    denominator /= nsd;
    return true;
}

bool fraction::setFraction(int num, int den) {
    if (den == 0) return false;
    numerator = num; denominator = den;
    return true;
}

fraction& fraction::operator=(const fraction& a) {
    setFraction(a.getNumerator(), a.getDenominator());
    return *this;
}

fraction& fraction::operator+=(const fraction& a) {
    int lcm  = std::lcm(denominator, a.getDenominator());
    int cit1 = lcm / denominator * numerator;
    int cit2 = lcm / a.getDenominator() * a.getNumerator();
    numerator   = cit1 + cit2;
    denominator = lcm;
    return *this;
}

fraction fraction::operator+(const fraction& a) const {
    fraction tmp(*this);     // kopírovací konstruktor
    tmp += a;                 // zavolá operator+=
    return tmp;
}

void fraction::swap(int& num, int& den) {
    int b = num; num = den; den = b;
}

ostream& operator<<(ostream& c, const fraction& d) {
    if ((d.getNumerator() < 0 && d.getDenominator() > 0) ||
        (d.getNumerator() > 0 && d.getDenominator() < 0))
        c << "-";
    c << abs(d.getNumerator()) << "/" << abs(d.getDenominator());
    return c;
}
Klíčové vzory
operator+= a operator+ — vzájemné delegování
operator+= modifikuje this a vrátí *this. operator+ je const, vytvoří kopii sebe sama (fraction tmp(*this)), zavolá tmp += a, a vrátí kopii. Díky tomu nemusíš psát logiku dvakrát.
std::gcd a std::lcm z <numeric> (C++17)
#include <numeric> → std::gcd(a, b) a std::lcm(a, b). Alternativně si můžeš napsat vlastní GCD (Eukleidův algoritmus): while(b) { a%=b; swap(a,b); } return a;
operator<< — výpis se správným znaménkem
Zlomek je záporný pokud mají čitatel a jmenovatel různá znaménka. Kód to testuje explicitně a vypíše "-", pak abs(num)/abs(den). Díky tomu se vyhne výpisům jako -3/-4.

fraction::fraction() : numerator(1), denominator(1) {} — dvojtečka za závorkami konstruktoru a seznam člen(hodnota). Efektivnější než přiřazení v těle konstruktoru (inicializuje místo assign).
