Klíčové algoritmy
convertToDec(string n, int s) — z libovolné do desítkové:
Jdi zleva: každá cifra × s^(pozice zprava). Pozice zprava = délka - i - 1.
cifry.find(n[i]) převede znakovou cifru ('A'=10, 'B'=11...) na číslo.
Příklad: "1010" v bin → 1×8 + 0×4 + 1×2 + 0×1 = 10
convertFromDec(unsigned int n, int s) — z desítkové do libovolné:
Opakuj: zbytek = n % s → přidej cifru, n = n / s. Na konci string otočíme pomocí reverse().
Speciální případ: n==0 → vrátí "0" (prázdná smyčka by vrátila "").

#ifndef SOUSTAVA_H
#define SOUSTAVA_H
#include <string>
#include <ostream>
using std::string;
using std::ostream;

class soustava {
public:
    soustava();
    soustava(string n, int s);
    soustava(const soustava& S);
    ~soustava() = default;

    string get_number(int s) const;
    bool   set_number(string n, int s);

private:
    unsigned int number;                          // VŽDY v desítkové!

    static bool        sIsValid(int s);
    static bool        nIsValid(string n, int s);
    static unsigned int convertToDec(string n, int s);
    static string       convertFromDec(unsigned int n, int s);
};

ostream& operator<<(ostream& os, const soustava& s);

#endif
#include "soustava.h"
#include <cmath>
#include <algorithm>

string cifry = "0123456789ABCDEF";       // globální lookup tabulka

soustava::soustava() : number(0) {}

soustava::soustava(string n, int s) {
    if (!set_number(n, s)) number = 0;
}

soustava::soustava(const soustava& S) {
    set_number(S.get_number(2), 2);          // zkopíruj přes binární reprezentaci
}

string soustava::get_number(int s) const {
    return sIsValid(s) ? convertFromDec(number, s) : "0";
}

bool soustava::set_number(string n, int s) {
    if (!nIsValid(n, s)) return false;
    number = convertToDec(n, s);
    return true;
}

bool soustava::sIsValid(int s) {
    return s >= 2 && s <= 16;
}

bool soustava::nIsValid(string n, int s) {
    if (!sIsValid(s)) return false;
    for (int i = 0; i < (int)n.length(); i++) {
        std::size_t found = cifry.find(n[i]);
        if (found == string::npos || found >= (size_t)s)
            return false;
    }
    return true;
}

unsigned int soustava::convertToDec(string n, int s) {
    unsigned int dec = 0;
    for (int i = 0; i < (int)n.length(); i++)
        dec += cifry.find(n[i]) * pow(s, n.length() - i - 1);
    return dec;
}

string soustava::convertFromDec(unsigned int n, int s) {
    string tmp = n == 0 ? "0" : "";
    while (n > 0) {
        tmp += cifry[n % s];              // cifry jsou pozpátku!
        n /= s;
    }
    std::reverse(tmp.begin(), tmp.end()); // otočit
    return tmp;
}

ostream& operator<<(ostream& os, const soustava& s) {
    os << "Cislo v dvojkove soustave :"    << s.get_number(2)  << std::endl;
    os << "Cislo v desitkove soustave :"   << s.get_number(10) << std::endl;
    os << "Cislo v sestnactkove soustave :" << s.get_number(16) << std::endl;
    os << "Cislo v osmickove soustave :"   << s.get_number(8)  << std::endl;
    return os;
}
Klíčové vzory
Datový model — vše interně jako unsigned int
Klíčová myšlenka: číslo se uloží jednou (v desítkové jako unsigned int number). get_number(s) ho vždy znovu převede do požadované soustavy. Díky tomu stačí dvě metody místo ukládání všech reprezentací.
cifry.find() — převod znak → číslo
String cifry = "0123456789ABCDEF" funguje jako lookup tabulka. cifry.find('A') vrátí 10, cifry.find('F') vrátí 15. cifry[10] vrátí 'A'. Elegantní obousměrný převod.
convertToDec — Hornerovo schéma vs. přímý výpočet
Kód používá pow(s, délka-i-1) z <cmath>. Alternativa je Hornerovo schéma (bez pow): dec = dec*s + cifra — efektivnější ale méně přímočaré. Oba přístupy fungují.
reverse() z <algorithm>
convertFromDec staví string pozpátku (nejnižší cifra první). Na konci std::reverse(tmp.begin(), tmp.end()) ho otočí. Nutno includovat <algorithm>.
