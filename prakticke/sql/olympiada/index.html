// POSTUP V PHPMyAdmin (krok za krokem)
  1. V levém panelu vyber databázi olympiada, klikni na Nová tabulka.
  2. Zadej název: sportoviste, počet sloupců: 6 → klikni Provést.
  3. Vyplň každý sloupec (formulář má zleva: Název / Typ / Délka–Hodnoty / Výchozí / Null / Index / A_I):
NázevTypDélka / HodnotyVýchozíNull ✔IndexA_I

NazevVARCHAR100————

  1. U idsportoviste – zaškrtni A_I → PHPMyAdmin navrhne PRIMARY index, potvrď.
  2. U Kapacita – v části Atributy zaškrtni UNSIGNED (záporné kapacity nedávají smysl).
  3. Po vyplnění všech sloupců klikni dole na Uložit.
SQL 
CREATE TABLE sportoviste (
  idsportoviste INT           NOT NULL AUTO_INCREMENT,
  Nazev         VARCHAR(100)  NOT NULL,
  Typ           ENUM('novostavba', 'rekonstrukce') DEFAULT NULL,
  Otevreni      DATE          DEFAULT NULL,
  Kapacita      INT(5) UNSIGNED DEFAULT NULL,
  Popis         TEXT          DEFAULT NULL,
  PRIMARY KEY (idsportoviste)
) ENGINE=InnoDB
  DEFAULT CHARSET=utf8
  COLLATE=utf8_czech_ci;
// KLÍČOVÉ VĚCI
  • ENUM('novostavba','rekonstrukce') – pouze dvě povolené hodnoty, nic jiného nelze vložit
  • INT(5) UNSIGNED – kladná celá čísla, závorka (5) je jen zobrazovací šířka v PHPMyAdmin, neomezuje hodnoty
  • TEXT – pro delší texty (až 65 535 znaků), vhodné pro popis areálu
  • DEFAULT NULL – nepovinné sloupce; pokud hodnotu nezadáš, uloží se NULL

d)
Vysocí sportovci – muži nad 190 cm, ženy nad 180 cm
Vypiš seznam všech sportovců vyšších než 190 cm a všech sportovkyň vyšších než 180 cm. Seřaď abecedně podle příjmení.
SQL 
SELECT
  CONCAT(s.jmeno, ' ', s.prijmeni)  AS `Jméno a příjmení`,
  st.nazev_statu                    AS `Stát`,
  s.pohlavi                        AS `Pohlaví`,
  CONCAT(ROUND(s.vyska * 100), ' cm') AS `Výška`

FROM sportovec s
JOIN stat st ON st.zkratka_statu = s.zkratka_statu

WHERE (s.pohlavi = 'muž'  AND s.vyska > 1.90)
   OR (s.pohlavi = 'žena' AND s.vyska > 1.80)

ORDER BY s.prijmeni ASC;
// JAK TO FUNGUJE
  • Sloupec vyska je uložen v metrech jako DECIMAL(3,2) – hodnoty jako 1.95 nebo 1.89
  • Podmínka (> 1.90) = "vyšší než 190 cm", závorky zajistí správné pořadí OR/AND
  • ROUND(vyska * 100) → z 1.89 udělá 189; CONCAT(..., ' cm') přidá jednotku
  • JOIN stat – připojíme tabulku států, abychom mohli zobrazit celý název státu

e)
Výsledky sedmiboje žen
Vypiš výsledky atletického závodu sedmiboj žen. Zobraz jméno sportovkyně, název státu, výkon a druh medaile. Seřaď podle umístění vzestupně.
SQL 
SELECT
  CONCAT(s.jmeno, ' ', s.prijmeni) AS `Jméno a příjmení`,
  st.nazev_statu                   AS `Stát`,
  ds.vykon                         AS `Výkon (b)`,
  u.medaile                       AS `Medaile`

FROM disciplina_sportovce ds
JOIN sportovec  s  ON s.idsportovec  = ds.idsportovec
JOIN disciplina d  ON d.iddiscipliny = ds.iddiscipliny
JOIN stat       st ON st.zkratka_statu = s.zkratka_statu
JOIN umisteni   u  ON u.poradi        = ds.umisteni

WHERE d.nazev_discipliny = 'sedmiboj'

ORDER BY ds.umisteni ASC;
// JAK TO FUNGUJE
  • Začínáme od disciplina_sportovce – ta propojuje sportovce s disciplínami a jejich výkonem
  • JOIN disciplina – umožní filtrovat podle nazev_discipliny = 'sedmiboj'
  • JOIN umisteni – tabulka přeloží číslo umístění (1,2,3) na druh medaile (zlatá, stříbrná, bronzová)
  • JOIN stat – potřebujeme celý název státu místo zkratky

Státy s více než jedním sportovcem – GROUP BY
Vypiš přehled počtu sportovců zastupujících jednotlivé země. Zobraz pouze státy s více než jedním sportovcem. Seřaď primárně podle počtu sportovců (sestupně), sekundárně podle názvu státu (abecedně).
SQL 
SELECT
  st.nazev_statu         AS `Stát`,
  COUNT(s.idsportovec)  AS `Počet sportovců`

FROM stat st
JOIN sportovec s ON s.zkratka_statu = st.zkratka_statu

GROUP BY st.zkratka_statu

HAVING COUNT(s.idsportovec) > 1

ORDER BY `Počet sportovců` DESC,
         st.nazev_statu    ASC;
// HAVING vs WHERE
  • GROUP BY zkratka_statu – seskupení podle zkratky (ne názvem – ten by mohl mít duplicity)
  • HAVING COUNT(...) > 1 – filtr po seskupení, takto filtrujeme výsledky agregací
  • WHERE by zde nefungovalo – to filtruje před seskupením, nemůže použít COUNT
  • Sekundární řazení nazev_statu ASC rozhodne pořadí při shodě počtu (Jamajka vs Spojené státy)

g)
Reprezentanti USA – odvětví, disciplína, medaile
Vypiš všechny representanty Spojených států amerických včetně názvu sportovního odvětví, označení disciplíny a získaného cenného kovu. Seřaď dle příjmení, pak dle umístění.
SQL 
SELECT
  CONCAT(s.jmeno, ' ', s.prijmeni) AS `Sportovec`,
  d.odvetvi_nazev_odvetvi           AS `Odvětví`,
  d.nazev_discipliny               AS `Disciplína`,
  u.medaile                       AS `Medaile`

FROM sportovec s
JOIN disciplina_sportovce ds ON ds.idsportovec  = s.idsportovec
JOIN disciplina           d  ON d.iddiscipliny  = ds.iddiscipliny
JOIN umisteni             u  ON u.poradi        = ds.umisteni

WHERE s.zkratka_statu = 'USA'

ORDER BY s.prijmeni ASC,
         ds.umisteni ASC;
// JAK TO FUNGUJE
  • Začínáme od sportovec a filtrujeme hned WHERE zkratka_statu = 'USA'
  • JOIN disciplina_sportovce – přidá výkon a umístění v každé disciplíně, kde sportovec závodil
  • JOIN disciplina – přidá název disciplíny a sportovního odvětví
  • JOIN umisteni – přeloží číslo umístění na medaili (zlatá/stříbrná/bronzová)
  • Phelps závodil ve dvou disciplínách → v tabulce se zobrazí jako dva řádky
