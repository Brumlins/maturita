// POSTUP V PHPMyAdmin (krok za krokem)

  2. Zadej název: nahravky, počet sloupců: 7 → klikni Provést.
  3. Vyplň každý sloupec (formulář má zleva: Název / Typ / Délka–Hodnoty / Výchozí / Null / Index / A_I):
NázevTypDélka / HodnotyVýchozíNull ✔IndexA_I

  1. U idnahravky – zaškrtni A_I → PHPMyAdmin navrhne PRIMARY index, potvrď.
  2. U kodek – vyber ENUM, do Délka/Hodnoty zadej: 'xvid','divx','wmv','jiný'
  3. U kodek – výchozí hodnotu nastav na as defined: xvid
  4. Po vyplnění všech sloupců klikni dole na Uložit.
SQL 
CREATE TABLE nahravky (
  idnahravky    INT          NOT NULL AUTO_INCREMENT,
  titul         VARCHAR(100)  NOT NULL,
  nazev_souboru VARCHAR(100)  NOT NULL,
  datum_ulozeni DATE          DEFAULT NULL,
  cas_ulozeni   TIME          DEFAULT NULL,
  kodek         ENUM('xvid','divx','wmv','jiný') NOT NULL DEFAULT 'xvid',
  velikost_mb   DECIMAL(6,1)  DEFAULT NULL,
  PRIMARY KEY (idnahravky)
) ENGINE=InnoDB
  DEFAULT CHARSET=utf8
  COLLATE=utf8_czech_ci;
// KLÍČOVÉ VĚCI
  • ENUM('xvid','divx','wmv','jiný') – jen povolené hodnoty, nic jiného nelze vložit
  • DEFAULT 'xvid' – pokud kodek nezadáš, automaticky se doplní xvid
  • DECIMAL(6,1) – číslo max. 6 číslic, z toho 1 za desetinnou čárkou → max. 99999.9 MB
  • TIME – formát HH:MM:SS, při vkládání stačí '22:30'
  • INT UNSIGNED – kladná celá čísla, vhodné pro ID (větší rozsah než INT)

Filmy natočené před rokem 1990 bez slova „Hrabal"
Vypiš všechny filmy natočené před rokem 1990, které nemají ve sloupci obsah slovo Hrabal. Seřaď od nejstaršího po nejmladší.
SQL 
SELECT
  nazev AS `Název`,
  rok   AS `Rok`

FROM filmy

WHERE rok < 1990
  AND (obsah NOT LIKE '%Hrabal%' OR obsah IS NULL)

ORDER BY rok ASC;
// JAK TO FUNGUJE
  • rok < 1990 – sloupec rok je typu YEAR, přímé porovnání s číslem funguje
  • NOT LIKE '%Hrabal%' – % je wildcard, hledáme "Hrabal" kdekoliv v textu
  • OR obsah IS NULL – bez toho by řádky s NULL obsah byly tichce vynechány

Historické, komediální a fantasy filmy z let 1981–2000
Vypiš filmy historického, komediálního a fantasy žánru natočené v letech 1981 až 2000. Seřaď primárně podle hodnocení (sestupně), sekundárně podle názvu (vzestupně).
SQL 
SELECT
  f.nazev       AS `Název`,
  z.nazev_zanru AS `Žánr`,
  f.rok         AS `Rok`,
  f.hodnoceni   AS `Hodnocení`

FROM filmy f
JOIN zanry z ON z.idzanr = f.idzanr

WHERE z.nazev_zanru IN ('historický', 'komedie', 'fantasy')
  AND f.rok BETWEEN 1981 AND 2000

ORDER BY f.hodnoceni DESC,
         f.nazev ASC;
// JAK TO FUNGUJE
  • JOIN zanry – propojíme filmy se žánry přes cizí klíč idzanr, abychom mohli filtrovat podle názvu žánru
  • IN ('historický', 'komedie', 'fantasy') – alternativa k třem podmínkám OR
  • BETWEEN 1981 AND 2000 – včetně obou krajních hodnot (1981 i 2000)
  • ORDER BY hodnoceni DESC, nazev ASC – nejdřív seřadí dle hodnocení, při shodě abecedně

Počet filmů podle režiséra – GROUP BY
K jednotlivým režisérům vypiš počet jimi natočených filmů v databázi. Seřaď podle počtu filmů (sestupně).
SQL 
SELECT
  CONCAT(u.jmeno, ' ', u.prijmeni) AS `Režisér`,
  COUNT(uvf.idfilmy)              AS `Počet filmů`

FROM umelci u
JOIN umelci_ve_filmech uvf ON uvf.idumelci = u.idumelci

WHERE uvf.role = 'režisér'

GROUP BY u.idumelci

ORDER BY `Počet filmů` DESC;
// KLÍČOVÉ KONCEPTY
  • WHERE role = 'režisér' – filtrujeme ještě před seskupením, bereme jen režiséry
  • GROUP BY u.idumelci – seskupujeme podle ID (ne jménem – jména se mohou opakovat)
  • COUNT(uvf.idfilmy) – počet řádků v umelci_ve_filmech pro daného umělce = počet jeho filmů
  • CONCAT(jmeno, ' ', prijmeni) – spojí jméno a příjmení do jednoho sloupce

Filmy s americkými a britskými herci narozenými po roce 1940
Vypiš filmy, v nichž se objevili američtí (USA) nebo britští (GBR) herci, kteří se narodili po roce 1940. Seřaď vzestupně podle data narození herce.
SQL 
SELECT
  f.nazev                                   AS `Název filmu`,
  CONCAT(u.jmeno, ' ', u.prijmeni)         AS `Herec`,
  s.nazev_statu                             AS `Stát`,
  DATE_FORMAT(u.narozeni, '%e. %c. %Y')    AS `Datum narození`

FROM filmy f
JOIN umelci_ve_filmech uvf ON uvf.idfilmy     = f.idfilmy
JOIN umelci u              ON u.idumelci     = uvf.idumelci
JOIN staty s               ON s.zkratka_statu = u.zkratka_statu

WHERE uvf.role = 'herec'
  AND u.zkratka_statu IN ('USA', 'GBR')
  AND YEAR(u.narozeni) > 1940

ORDER BY u.narozeni ASC;
// JAK TO FUNGUJE
  • JOIN staty – přidáme tabulku států, abychom mohli zobrazit celý název státu
  • WHERE role = 'herec' – bereme jen herce, ne režiséry nebo scenáristy
  • YEAR(narozeni) > 1940 – „po roce 1940" znamená rok > 1940, John Hurt (1940) tím vypadne
  • DATE_FORMAT – %e = den bez nuly, %c = měsíc bez nuly, %Y = rok

Smazání umělců narozených v únoru nebo bez data narození
Smaž z databáze všechny umělce, kteří se narodili v únoru, nebo u nichž datum narození není uvedeno. Zároveň zajisti, aby byly odstraněny i jejich záznamy v tabulce umelci_ve_filmech.
// PROČ NESTAČÍ JEDEN DELETE
  • Tabulka umelci_ve_filmech má cizí klíč na umelci s nastavením ON DELETE NO ACTION
  • To znamená: MySQL odmítne smazat umělce, pokud na něj existuje odkaz ve vázané tabulce
  • Řešení: nejdřív smažeme záznamy z umelci_ve_filmech, pak teprve z umelci
SQL 
-- 1. nejdříve odstraníme záznamy ve vázané tabulce
DELETE FROM umelci_ve_filmech
WHERE idumelci IN (
  SELECT idumelci FROM umelci
  WHERE MONTH(narozeni) = 2 OR narozeni IS NULL
);

-- 2. pak smažeme samotné umělce
DELETE FROM umelci
WHERE MONTH(narozeni) = 2 OR narozeni IS NULL;
// CO SE SMAŽE
  • MONTH(narozeni) = 2 – Jiří Menzel (23. 2.), Jaromír Hanzlík (16. 2.), Otakar Vávra (28. 2.)
  • narozeni IS NULL – Steve Zaillian (datum není uvedeno)
  • Celkem: 4 umělci a 8 jejich záznamů v tabulce umelci_ve_filmech
i)
Přejmenování sloupce delka na stopaz
V tabulce filmy přejmenuj sloupec delka na stopaz. Datový typ a výchozí hodnota zůstávají stejné.
SQL 
ALTER TABLE filmy
  CHANGE delka stopaz INT(3) DEFAULT NULL;
// POZNÁMKA
  • CHANGE vyžaduje uvést i datový typ – píšeš starý název, nový název a pak typ
  • Alternativa v MySQL 8+: ALTER TABLE filmy RENAME COLUMN delka TO stopaz;
j)
Smazání tabulky nahravky a celé databáze
Nejdřív odstraň tabulku nahravky, pak úplně smaž celou databázi filmoteka.
SQL 
-- odstraní tabulku nahravky
DROP TABLE nahravky;

-- odstraní celou databázi se všemi tabulkami
DROP DATABASE filmoteka;
// POZOR
  • DROP TABLE smaže tabulku včetně všech dat, nelze vrátit
  • DROP DATABASE smaže celou databázi se všemi tabulkami a daty
  • Pořadí záleží – nejdřív tabulka, pak DB. Opačně by DROP TABLE selhal, protože DB by neexistovala
