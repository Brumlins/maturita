Filtrace a zpracování textu

📌 Klíčové body

  • grep (-i, -r, -v, -E) – hledání vzorů
  • sed 's/staré/nové/g' – nahrazování, mazání řádků
  • awk '{print $1}' – zpracování sloupců, suma
  • sort, uniq, cut, tr, wc, head, tail, roury

grep – hledání vzorů

# Základní použití
grep "vzor" soubor.txt
grep -i "vzor" soubor.txt         # case-insensitive
grep -r "vzor" /etc/              # rekurzivně
grep -n "vzor" soubor.txt         # zobraz čísla řádků
grep -v "vzor" soubor.txt         # inverzní – řádky BEZ vzoru
grep -c "vzor" soubor.txt         # počet shod
grep -l "vzor" *.txt              # jen jména souborů

# Rozšířené regulární výrazy
grep -E "^[0-9]+$" soubor.txt     # jen čísla
grep -E "error|warning" app.log
grep -P "\d{3}-\d{4}" data.txt    # Perl regex (telefonní číslo)

sed – proudový editor

# Nahrazení (s = substitute)
sed 's/staré/nové/' soubor.txt          # první výskyt na řádku
sed 's/staré/nové/g' soubor.txt         # všechny výskyty
sed 's/staré/nové/gi' soubor.txt        # case-insensitive + global
sed -i 's/foo/bar/g' soubor.txt         # IN-PLACE úprava souboru!

# Mazání řádků
sed '5d' soubor.txt                     # smaž řádek 5
sed '/vzor/d' soubor.txt                # smaž řádky s vzorem
sed '1,3d' soubor.txt                   # smaž řádky 1–3

# Tisk specifických řádků
sed -n '5,10p' soubor.txt               # tiskni řádky 5–10
sed -n '/START/,/END/p' soubor.txt      # mezi vzory

# Přidání řádku
sed '2a\Nový řádek' soubor.txt          # za řádek 2
sed '2i\Nový řádek' soubor.txt          # před řádek 2

awk – zpracování sloupců

# Základy: awk pracuje s poli (sloupci) oddělených mezerou/FS
awk '{print $1}' soubor.txt             # první sloupec
awk '{print $1, $3}' soubor.txt         # 1. a 3. sloupec
awk '{print NR, $0}' soubor.txt         # číslo řádku + řádek

# Podmínky
awk '$3 > 100 {print $1, $3}' data.txt
awk '/vzor/ {print $2}' soubor.txt

# Oddělovač
awk -F: '{print $1, $7}' /etc/passwd    # uživatel:shell
awk -F',' '{print $2}' data.csv

# Suma
awk '{sum += $3} END {print "Suma:", sum}' data.txt

# Více operací
awk 'BEGIN{print "Start"} {print $1} END{print "Konec"}' f.txt

Další filtrovací nástroje

# head / tail
head -5 soubor.txt          # prvních 5 řádků
tail -10 soubor.txt         # posledních 10 řádků
tail -f /var/log/syslog     # live sledování logu

# sort
sort soubor.txt             # abecedně
sort -n čísla.txt           # numericky
sort -r soubor.txt          # reverzní
sort -k2 data.txt           # podle 2. sloupce
sort -u soubor.txt          # unikátní (odstraní duplicity)

# uniq (potřebuje seřazený vstup)
uniq soubor.txt             # odstraní sousední duplicity
uniq -c soubor.txt          # počty výskytů
sort soubor.txt | uniq -d   # jen duplicitní řádky

# cut – vyříznutí sloupců
cut -d: -f1,7 /etc/passwd   # oddělovač : sloupce 1 a 7
cut -c1-5 soubor.txt        # znaky 1–5

# tr – překlad znaků
echo "hello" | tr 'a-z' 'A-Z'   # převod na velká
echo "hello world" | tr -d ' '   # smaž mezery
cat soubor.txt | tr -s ' '       # zredukuj vícenásobné mezery

# wc – počítání
wc -l soubor.txt           # počet řádků
wc -w soubor.txt           # počet slov
wc -c soubor.txt           # počet bytů

Roury a přesměrování

# Přesměrování výstupu
příkaz > soubor.txt         # přepis
příkaz >> soubor.txt        # přidání
příkaz 2> chyby.txt         # stderr
příkaz &> vše.txt           # stdout + stderr

# Roury (pipeline)
cat /etc/passwd | grep bash | cut -d: -f1 | sort

# Kombinace
ps aux | grep nginx | grep -v grep | awk '{print $2}'

# xargs – předání jako argumenty
find . -name "*.log" | xargs rm
echo "soubor.txt" | xargs cat
