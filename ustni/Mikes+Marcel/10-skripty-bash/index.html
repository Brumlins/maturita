Bash skripty – základy

📌 Klíčové body

  • Shebang #!/bin/bash, proměnné, $1..$N, $@, $#
  • if/elif/else, testovací operátory (-f, -d, -z, -eq)
  • for (seznam, rozsah, C-style), while, until
  • Funkce, local proměnné, return kód

Základy Bash skriptování

#!/bin/bash
# Shebang – říká shellu, který interpret použít

# Proměnné (bez mezer kolem =!)
jmeno="Jan"
vek=25
echo "Jmenuji se $jmeno a je mi $vek let."
echo "Délka řetězce: ${#jmeno}"

# Vstupy
read -p "Zadej jméno: " jmeno
echo "Ahoj, $jmeno!"

# Pozicní parametry
echo "Skript: $0"
echo "Argument 1: $1"
echo "Počet argumentů: $#"
echo "Všechny argumenty: $@"
echo "Exit code: $?"

Podmínky – if/elif/else

#!/bin/bash
cislo=42

if [ "$cislo" -gt 100 ]; then
    echo "Velké číslo"
elif [ "$cislo" -gt 50 ]; then
    echo "Středně velké"
else
    echo "Malé číslo"
fi

# Test souborů
if [ -f "/etc/passwd" ]; then echo "Soubor existuje"; fi
if [ -d "/home" ];       then echo "Je to adresář"; fi
if [ -r "/etc/passwd" ]; then echo "Čitelný"; fi
if [ -z "$var" ];        then echo "Prázdná proměnná"; fi
if [ -n "$var" ];        then echo "Neprázdná proměnná"; fi

# Srovnání řetězců
if [ "$str" = "hello" ]; then echo "Rovná se"; fi
if [ "$str" != "bye" ];  then echo "Nerovná se"; fi

# Logické operátory
if [ $a -gt 0 ] && [ $b -gt 0 ]; then echo "Obě kladné"; fi
if [[ $a -gt 0 && $b -gt 0 ]];   then echo "Obě kladné (novější)"; fi

Cykly

# For – iterace přes seznam
for i in 1 2 3 4 5; do
    echo "Číslo: $i"
done

# For – rozsah
for i in {1..10}; do echo $i; done
for i in {0..20..2}; do echo $i; done   # po 2

# For – C-style
for (( i=0; i<10; i++ )); do
    echo "$i"
done

# For – soubory
for soubor in /tmp/*.log; do
    echo "Zpracovávám: $soubor"
done

# While
while [ "$odpoved" != "ano" ]; do
    read -p "Pokračovat? " odpoved
done

# Until
until ping -c1 google.com &>/dev/null; do
    echo "Čekám na síť..."
    sleep 2
done

Funkce

#!/bin/bash

pozdrav() {
    local jmeno="$1"       # local = lokální proměnná
    echo "Ahoj, $jmeno!"
    return 0               # exit code funkce
}

pozdrav "Světe"

# Funkce s výstupem
vypocet() {
    echo $(( $1 + $2 ))
}
vysledek=$(vypocet 5 3)
echo "5 + 3 = $vysledek"

Práce s řetězci a aritmetika

# Aritmetika
echo $(( 5 + 3 ))
vysledek=$(( 10 * 4 - 2 ))

# Substituce v řetězcích
str="Hello World"
echo "${str,,}"          # malá písmena: hello world
echo "${str^^}"          # velká písmena: HELLO WORLD
echo "${str/World/Bash}" # nahraď: Hello Bash
echo "${str:0:5}"        # substring: Hello
